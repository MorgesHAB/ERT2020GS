48 uint16_t RHcrc16_update(uint16_t crc, uint8_t a)
49 {
50  int i;
51 
52  crc ^= a;
53  for (i = 0; i < 8; ++i)
54  {
55  if (crc & 1)
56  crc = (crc >> 1) ^ 0xA001;
57  else
58  crc = (crc >> 1);
59  }
60  return crc;
61 }
