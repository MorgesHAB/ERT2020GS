75 bool RH_RF24::init()
76 {
77  if (!RHSPIDriver::init())
78  return false;
79 
80  // Determine the interrupt number that corresponds to the interruptPin
81  int interruptNumber = digitalPinToInterrupt(_interruptPin);
82  if (interruptNumber == NOT_AN_INTERRUPT)
83  return false;
84 
85  // Initialise the radio
86  power_on_reset();
87  cmd_clear_all_interrupts();
88  // Here we use a configuration generated by the Silicon Las Wireless Development Suite
89  // in radio_config_Si4460.h
90  // WE override a few things later that we ned to be sure of.
91  configure(RFM26_CONFIGURATION_DATA);
92 
93  // Get the device type and check it
94  // This also tests whether we are really connected to a device
95  uint8_t buf[8];
96  if (!command(RH_RF24_CMD_PART_INFO, 0, 0, buf, sizeof(buf)))
97  return false; // SPI error? Not connected?
98  _deviceType = (buf[1] << 8) | buf[2];
99  // Check PART to be either 0x4460, 0x4461, 0x4463, 0x4464
100  if (_deviceType != 0x4460 &&
101  _deviceType != 0x4461 &&
102  _deviceType != 0x4463 &&
103  _deviceType != 0x4464)
104  return false; // Unknown radio type, or not connected
105 
106  // Add by Adrien van den Bossche <vandenbo@univ-tlse2.fr> for Teensy
107  // ARM M4 requires the below. else pin interrupt doesn't work properly.
108  // On all other platforms, its innocuous, belt and braces
109  pinMode(_interruptPin, INPUT);
110 
111  // Set up interrupt handler
112  // Since there are a limited number of interrupt glue functions isr*() available,
113  // we can only support a limited number of devices simultaneously
114  // ON some devices, notably most Arduinos, the interrupt pin passed in is actuallt the
115  // interrupt number. You have to figure out the interruptnumber-to-interruptpin mapping
116  // yourself based on knwledge of what Arduino board you are running on.
117  if (_myInterruptIndex == 0xff)
118  {
119  // First run, no interrupt allocated yet
120  if (_interruptCount <= RH_RF24_NUM_INTERRUPTS)
121  _myInterruptIndex = _interruptCount++;
122  else
123  return false; // Too many devices, not enough interrupt vectors
124  }
125  _deviceForInterrupt[_myInterruptIndex] = this;
126  if (_myInterruptIndex == 0)
127  attachInterrupt(interruptNumber, isr0, FALLING);
128  else if (_myInterruptIndex == 1)
129  attachInterrupt(interruptNumber, isr1, FALLING);
130  else if (_myInterruptIndex == 2)
131  attachInterrupt(interruptNumber, isr2, FALLING);
132  else
133  return false; // Too many devices, not enough interrupt vectors
134 
135  // Ensure we get the interrupts we need, irrespective of whats in the radio_config
136  uint8_t int_ctl[] = {RH_RF24_MODEM_INT_STATUS_EN | RH_RF24_PH_INT_STATUS_EN, 0xff, 0xff, 0x00 };
137  set_properties(RH_RF24_PROPERTY_INT_CTL_ENABLE, int_ctl, sizeof(int_ctl));
138 
139  // RSSI Latching should be configured in MODEM_RSSI_CONTROL in radio_config
140 
141  // PKT_TX_THRESHOLD and PKT_RX_THRESHOLD should be set to about 0x30 in radio_config
142 
143  // Configure important RH_RF24 registers
144  // Here we set up the standard packet format for use by the RH_RF24 library:
145  // We will use FIFO Mode, with automatic packet generation
146  // We have 2 fields:
147  // Field 1 contains only the (variable) length of field 2, with CRC
148  // Field 2 contains the variable length payload and the CRC
149  // Hmmm, having no CRC on field 1 and CRC on field 2 causes CRC errors when resetting after an odd
150  // number of packets! Anyway its prob a good thing at the cost of some airtime.
151  // Hmmm, enabling WHITEN stops it working!
152  uint8_t pkt_config1[] = { 0x00 };
153  set_properties(RH_RF24_PROPERTY_PKT_CONFIG1, pkt_config1, sizeof(pkt_config1));
154 
155  uint8_t pkt_len[] = { 0x02, 0x01, 0x00 };
156  set_properties(RH_RF24_PROPERTY_PKT_LEN, pkt_len, sizeof(pkt_len));
157 
158  uint8_t pkt_field1[] = { 0x00, 0x01, 0x00, RH_RF24_FIELD_CONFIG_CRC_START | RH_RF24_FIELD_CONFIG_SEND_CRC | RH_RF24_FIELD_CONFIG_CHECK_CRC | RH_RF24_FIELD_CONFIG_CRC_ENABLE };
159  set_properties(RH_RF24_PROPERTY_PKT_FIELD_1_LENGTH_12_8, pkt_field1, sizeof(pkt_field1));
160 
161  uint8_t pkt_field2[] = { 0x00, sizeof(_buf), 0x00, RH_RF24_FIELD_CONFIG_CRC_START | RH_RF24_FIELD_CONFIG_SEND_CRC | RH_RF24_FIELD_CONFIG_CHECK_CRC | RH_RF24_FIELD_CONFIG_CRC_ENABLE };
162  set_properties(RH_RF24_PROPERTY_PKT_FIELD_2_LENGTH_12_8, pkt_field2, sizeof(pkt_field2));
163 
164  // Clear all other fields so they are never used, irrespective of the radio_config
165  uint8_t pkt_fieldn[] = { 0x00, 0x00, 0x00, 0x00 };
166  set_properties(RH_RF24_PROPERTY_PKT_FIELD_3_LENGTH_12_8, pkt_fieldn, sizeof(pkt_fieldn));
167  set_properties(RH_RF24_PROPERTY_PKT_FIELD_4_LENGTH_12_8, pkt_fieldn, sizeof(pkt_fieldn));
168  set_properties(RH_RF24_PROPERTY_PKT_FIELD_5_LENGTH_12_8, pkt_fieldn, sizeof(pkt_fieldn));
169 
170  // The following can be changed later by the user if necessary.
171  // Set up default configuration
172  setCRCPolynomial(CRC_16_IBM);
173  uint8_t syncwords[] = { 0x2d, 0xd4 };
174  setSyncWords(syncwords, sizeof(syncwords)); // Same as RF22's
175  // Reasonably fast and reliable default speed and modulation
176  setModemConfig(GFSK_Rb5Fd10);
177  // 3 would be sufficient, but this is the same as RF22's
178  // actualy, 4 seems to work much better for some modulations
179  setPreambleLength(4);
180  // An innocuous ISM frequency, same as RF22's
181  setFrequency(434.0);
182  // About 2.4dBm on RFM24:
183  setTxPower(0x10);
184 
185  return true;
186 }

188 // C++ level interrupt handler for this instance
189 void RH_RF24::handleInterrupt()
190 {
191  uint8_t status[8];
192  command(RH_RF24_CMD_GET_INT_STATUS, NULL, 0, status, sizeof(status));
193 
194  // Decode and handle the interrupt bits we are interested in
195 // if (status[0] & RH_RF24_INT_STATUS_CHIP_INT_STATUS)
196  if (status[0] & RH_RF24_INT_STATUS_MODEM_INT_STATUS)
197  {
198 // if (status[4] & RH_RF24_INT_STATUS_INVALID_PREAMBLE)
199  if (status[4] & RH_RF24_INT_STATUS_INVALID_SYNC)
200  {
201  // After INVALID_SYNC, sometimes the radio gets into a silly state and subsequently reports it for every packet
202  // Need to reset the radio and clear the RX FIFO, cause sometimes theres junk there too
203  _mode = RHModeIdle;
204  clearRxFifo();
205  clearBuffer();
206  }
207  }
208  if (status[0] & RH_RF24_INT_STATUS_PH_INT_STATUS)
209  {
210  if (status[2] & RH_RF24_INT_STATUS_CRC_ERROR)
211  {
212  // CRC Error
213  // Radio automatically went to _idleMode
214  _mode = RHModeIdle;
215  _rxBad++;
216 
217  clearRxFifo();
218  clearBuffer();
219  }
220  if (status[2] & RH_RF24_INT_STATUS_PACKET_SENT)
221  {
222  _txGood++;
223  // Transmission does not automatically clear the tx buffer.
224  // Could retransmit if we wanted
225  // RH_RF24 configured to transition automatically to Idle after packet sent
226  _mode = RHModeIdle;
227  clearBuffer();
228  }
229  if (status[2] & RH_RF24_INT_STATUS_PACKET_RX)
230  {
231  // A complete message has been received with good CRC
232  // Get the RSSI, configured to latch at sync detect in radio_config
233  uint8_t modem_status[6];
234  command(RH_RF24_CMD_GET_MODEM_STATUS, NULL, 0, modem_status, sizeof(modem_status));
235  _lastRssi = modem_status[3];
236  _lastPreambleTime = millis();
237 
238  // Save it in our buffer
239  readNextFragment();
240  // And see if we have a valid message
241  validateRxBuf();
242  // Radio will have transitioned automatically to the _idleMode
243  _mode = RHModeIdle;
244  }
245  if (status[2] & RH_RF24_INT_STATUS_TX_FIFO_ALMOST_EMPTY)
246  {
247  // TX FIFO almost empty, maybe send another chunk, if there is one
248  sendNextFragment();
249  }
250  if (status[2] & RH_RF24_INT_STATUS_RX_FIFO_ALMOST_FULL)
251  {
252  // Some more data to read, get it
253  readNextFragment();
254  }
255  }
256 }

321 {
322  if (!available())
323  return false;

326  {

337 bool RH_RF24::send(const uint8_t* data, uint8_t len)
338 {
339  if (len > RH_RF24_MAX_MESSAGE_LEN)
340  return false;
341 
342  waitPacketSent(); // Make sure we dont interrupt an outgoing message
343  setModeIdle(); // Prevent RX while filling the fifo
344 
345  // Put the payload in the FIFO
346  // First the length in fixed length field 1. This wont appear in the receiver fifo since
347  // we have turned off IN_FIFO in PKT_LEN
348  _buf[0] = len + RH_RF24_HEADER_LEN;
349  // Now the rest of the payload in variable length field 2
350  // First the headers

384 void RH_RF24::sendNextFragment()
385 {
386  if (_txBufSentIndex < _bufLen)
387  {
388  // Some left to send?
389  uint8_t len = _bufLen - _txBufSentIndex;
390  // But dont send too much, see how much room is left
391  uint8_t fifo_info[2];
392  command(RH_RF24_CMD_FIFO_INFO, NULL, 0, fifo_info, sizeof(fifo_info));
393  // fifo_info[1] is space left in TX FIFO
394  if (len > fifo_info[1])
395  len = fifo_info[1];
396 
397  writeTxFifo(_buf + _txBufSentIndex, len);
398  _txBufSentIndex += len;
399  }

402 void RH_RF24::readNextFragment()
403 {
404  // Get the packet length from the RX FIFO length
405  uint8_t fifo_info[1];
406  command(RH_RF24_CMD_FIFO_INFO, NULL, 0, fifo_info, sizeof(fifo_info));
407  uint8_t fifo_len = fifo_info[0];
408 
409  // Check for overflow
410  if ((_bufLen + fifo_len) > sizeof(_buf))
411  {
412  // Overflow pending
413  _rxBad++;
414  setModeIdle();
415  clearRxFifo();
416  clearBuffer();
417  return;
418  }
419  // So we have room
420  // Now read the fifo_len bytes from the RX FIFO
421  // This is different to command() since we dont wait for CTS
422  digitalWrite(_slaveSelectPin, LOW);
423  _spi.transfer(RH_RF24_CMD_RX_FIFO_READ);
424  uint8_t* p = _buf + _bufLen;
425  uint8_t l = fifo_len;
426  while (l--)
427  *p++ = _spi.transfer(0);
428  digitalWrite(_slaveSelectPin, HIGH);
429  _bufLen += fifo_len;
430 }

566 bool RH_RF24::setFrequency(float centre, float afcPullInRange)
567 {
568  // See Si446x Data Sheet section 5.3.1
569  // Also the Si446x PLL Synthesizer / VCO_CNT Calculator Rev 0.4
570  uint8_t outdiv;
571  uint8_t band;
572  if (_deviceType == 0x4460 ||
573  _deviceType == 0x4461 ||
574  _deviceType == 0x4463)
575  {
576  // Non-continuous frequency bands
577  if (centre <= 1050.0 && centre >= 850.0)
578  outdiv = 4, band = 0;
579  else if (centre <= 525.0 && centre >= 425.0)
580  outdiv = 8, band = 2;
581  else if (centre <= 350.0 && centre >= 284.0)
582  outdiv = 12, band = 3;
583  else if (centre <= 175.0 && centre >= 142.0)
584  outdiv = 24, band = 5;
585  else
586  return false;
587  }
588  else
589  {
590  // 0x4464
591  // Continuous frequency bands
592  if (centre <= 960.0 && centre >= 675.0)
593  outdiv = 4, band = 1;
594  else if (centre < 675.0 && centre >= 450.0)
595  outdiv = 6, band = 2;
596  else if (centre < 450.0 && centre >= 338.0)
597  outdiv = 8, band = 3;
598  else if (centre < 338.0 && centre >= 225.0)
599  outdiv = 12, band = 4;
600  else if (centre < 225.0 && centre >= 169.0)
601  outdiv = 16, band = 4;
602  else if (centre < 169.0 && centre >= 119.0)
603  outdiv = 24, band = 5;
604  else
605  return false;
606  }
607 
608  // Set the MODEM_CLKGEN_BAND (not documented)
609  uint8_t modem_clkgen[] = { band+8 };
610  if (!set_properties(RH_RF24_PROPERTY_MODEM_CLKGEN_BAND, modem_clkgen, sizeof(modem_clkgen)))
611  return false;
612 
613  centre *= 1000000.0; // Convert to Hz
614 
615  // Now generate the RF frequency properties
616  // Need the Xtal/XO freq from the radio_config file:
617  uint32_t xtal_frequency[1] = RADIO_CONFIGURATION_DATA_RADIO_XO_FREQ;
618  unsigned long f_pfd = 2 * xtal_frequency[0] / outdiv;
619  unsigned int n = ((unsigned int)(centre / f_pfd)) - 1;
620  float ratio = centre / (float)f_pfd;
621  float rest = ratio - (float)n;
622  unsigned long m = (unsigned long)(rest * 524288UL);
623  unsigned int m2 = m / 0x10000;
624  unsigned int m1 = (m - m2 * 0x10000) / 0x100;
625  unsigned int m0 = (m - m2 * 0x10000 - m1 * 0x100);
626 
627  // PROP_FREQ_CONTROL_GROUP
628  uint8_t freq_control[] = { n, m2, m1, m0 };
629  return set_properties(RH_RF24_PROPERTY_FREQ_CONTROL_INTE, freq_control, sizeof(freq_control));
630 }

726 // Caution: There was a bug in A1 hardware that will not handle 1 byte commands.
727 bool RH_RF24::command(uint8_t cmd, const uint8_t* write_buf, uint8_t write_len, uint8_t* read_buf, uint8_t read_len)
728 {
729  bool done = false;
730 
731  ATOMIC_BLOCK_START;
732  // First send the command
733  digitalWrite(_slaveSelectPin, LOW);
734  _spi.transfer(cmd);
735 
736  // Now write any write data
737  if (write_buf && write_len)
738  {
739  while (write_len--)
740  _spi.transfer(*write_buf++);
741  }
742  // Sigh, the RFM26 at least has problems if we deselect too quickly :-(
743  // Innocuous timewaster:
744  digitalWrite(_slaveSelectPin, LOW);
745  // And finalise the command
746  digitalWrite(_slaveSelectPin, HIGH);
747 
748  uint16_t count; // Number of times we have tried to get CTS
749  for (count = 0; !done && count < RH_RF24_CTS_RETRIES; count++)
750  {
751  // Wait for the CTS
752  digitalWrite(_slaveSelectPin, LOW);
753 
754  _spi.transfer(RH_RF24_CMD_READ_BUF);
755  if (_spi.transfer(0) == RH_RF24_REPLY_CTS)
756  {
757  // Now read any expected reply data
758  if (read_buf && read_len)
759  {
760  while (read_len--)
761  {
762  *read_buf++ = _spi.transfer(0);
763  }
764  }
765  done = true;
766  }
767  // Sigh, the RFM26 at least has problems if we deselect too quickly :-(
768  // Innocuous timewaster:
769  digitalWrite(_slaveSelectPin, LOW);
770  // Finalise the read
771  digitalWrite(_slaveSelectPin, HIGH);
772  }
773  ATOMIC_BLOCK_END;
774  return done; // False if too many attempts at CTS
775 }
