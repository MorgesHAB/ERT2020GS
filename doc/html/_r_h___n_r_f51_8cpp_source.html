17 bool RH_NRF51::init()
18 {
19  // Enable the High Frequency clock to the system as a whole
20  NRF_CLOCK->EVENTS_HFCLKSTARTED = 0;
21  NRF_CLOCK->TASKS_HFCLKSTART = 1;
22  /* Wait for the external oscillator to start up */
23  while (NRF_CLOCK->EVENTS_HFCLKSTARTED == 0) { }
24 
25  // Enables the DC/DC converter when the radio is enabled. Need this!
26  NRF_POWER->DCDCEN = 0x00000001;
27 
28  // Disable and reset the radio
29  NRF_RADIO->POWER = RADIO_POWER_POWER_Disabled;
30  NRF_RADIO->POWER = RADIO_POWER_POWER_Enabled;
31  NRF_RADIO->EVENTS_DISABLED = 0;
32  NRF_RADIO->TASKS_DISABLE = 1;
33  // Wait until we are in DISABLE state
34  while (NRF_RADIO->EVENTS_DISABLED == 0) {}
35 
36  // Physical on-air address is set in PREFIX0 + BASE0 by setNetworkAddress
37  NRF_RADIO->TXADDRESS = 0x00; // Use logical address 0 (PREFIX0 + BASE0)
38  NRF_RADIO->RXADDRESSES = 0x01; // Enable reception on logical address 0 (PREFIX0 + BASE0)
39 
40  // Configure the CRC
41  NRF_RADIO->CRCCNF = (RADIO_CRCCNF_LEN_Two << RADIO_CRCCNF_LEN_Pos); // Number of checksum bits
42  NRF_RADIO->CRCINIT = 0xFFFFUL; // Initial value
43  NRF_RADIO->CRCPOLY = 0x11021UL; // CRC poly: x^16+x^12^x^5+1
44 
45  // These shorts will make the radio transition from Ready to Start to Disable automatically
46  // for both TX and RX, which makes for much shorter on-air times
47  NRF_RADIO->SHORTS = (RADIO_SHORTS_READY_START_Enabled << RADIO_SHORTS_READY_START_Pos)
48  | (RADIO_SHORTS_END_DISABLE_Enabled << RADIO_SHORTS_END_DISABLE_Pos);
49 
50  NRF_RADIO->PCNF0 = ((8 << RADIO_PCNF0_LFLEN_Pos) & RADIO_PCNF0_LFLEN_Msk); // Payload length in bits
51 
52  // Make sure we are powered down
53  setModeIdle();
54 
55  // Set a default network address
56  uint8_t default_network_address[] = {0xE7, 0xE7, 0xE7, 0xE7, 0xE7};
57  setNetworkAddress(default_network_address, sizeof(default_network_address));
58 
59  setChannel(2); // The default, in case it was set by another app without powering down
60  setRF(RH_NRF51::DataRate2Mbps, RH_NRF51::TransmitPower0dBm);
61 
62  return true;
63 }

71 bool RH_NRF51::setNetworkAddress(uint8_t* address, uint8_t len)
72 {
73  if (len < 3 || len > 5)
74  return false;
75 
76  // First byte is the prefix, remainder are base
77  NRF_RADIO->PREFIX0 = ((address[0] << RADIO_PREFIX0_AP0_Pos) & RADIO_PREFIX0_AP0_Msk);
78  uint32_t base;
79  memcpy(&base, address+1, len-1);
80  NRF_RADIO->BASE0 = base;
81 
82  NRF_RADIO->PCNF1 = (
83  (((sizeof(_buf)) << RADIO_PCNF1_MAXLEN_Pos) & RADIO_PCNF1_MAXLEN_Msk) // maximum length of payload
84  | (((0UL) << RADIO_PCNF1_STATLEN_Pos) & RADIO_PCNF1_STATLEN_Msk) // expand the payload with 0 bytes
85  | (((len-1) << RADIO_PCNF1_BALEN_Pos) & RADIO_PCNF1_BALEN_Msk)); // base address length in number of bytes.
86 
87  return true;
88 }

90 bool RH_NRF51::setRF(DataRate data_rate, TransmitPower power)
91 {
92  uint8_t mode;
93  uint8_t p;
94 
95  if (data_rate == DataRate2Mbps)
96  mode = RADIO_MODE_MODE_Nrf_2Mbit;
97  else if (data_rate == DataRate1Mbps)
98  mode = RADIO_MODE_MODE_Nrf_1Mbit;
99  else if (data_rate == DataRate250kbps)
100  mode = RADIO_MODE_MODE_Nrf_250Kbit;
101  else
102  return false;// Invalid
103 
104  if (power == TransmitPower4dBm)
105  p = RADIO_TXPOWER_TXPOWER_Pos4dBm;
106  else if (power == TransmitPower0dBm)
107  p = RADIO_TXPOWER_TXPOWER_0dBm;
108  else if (power == TransmitPowerm4dBm)
109  p = RADIO_TXPOWER_TXPOWER_Neg4dBm;
110  else if (power == TransmitPowerm8dBm)
111  p = RADIO_TXPOWER_TXPOWER_Neg8dBm;
112  else if (power == TransmitPowerm12dBm)
113  p = RADIO_TXPOWER_TXPOWER_Neg12dBm;
114  else if (power == TransmitPowerm16dBm)
115  p = RADIO_TXPOWER_TXPOWER_Neg16dBm;
116  else if (power == TransmitPowerm20dBm)
117  p = RADIO_TXPOWER_TXPOWER_Neg20dBm;
118  else if (power == TransmitPowerm30dBm)
119  p = RADIO_TXPOWER_TXPOWER_Neg30dBm;
120  else
121  return false; // Invalid
122 
123 
124  NRF_RADIO->TXPOWER = ((p << RADIO_TXPOWER_TXPOWER_Pos) & RADIO_TXPOWER_TXPOWER_Msk);
125  NRF_RADIO->MODE = ((mode << RADIO_MODE_MODE_Pos) & RADIO_MODE_MODE_Msk);
126 
127  return true;
128 }

165 bool RH_NRF51::send(const uint8_t* data, uint8_t len)
166 {
167  if (len > RH_NRF51_MAX_MESSAGE_LEN)
168  return false;
169  // Set up the headers
170  _buf[0] = len + RH_NRF51_HEADER_LEN;
171  _buf[1] = _txHeaderTo;
172  _buf[2] = _txHeaderFrom;
173  _buf[3] = _txHeaderId;
174  _buf[4] = _txHeaderFlags;
175  memcpy(_buf+RH_NRF51_HEADER_LEN+1, data, len);
176 
177  _rxBufValid = false;
178  setModeTx();
179  // Radio will return to Disabled state after transmission is complete
180  _txGood++;
181  return true;
182 }

184 bool RH_NRF51::waitPacketSent()
185 {
186  // If we are not currently in transmit mode, there is no packet to wait for
187  if (_mode != RHModeTx)
188  return false;
189 
190  // When the Disabled event occurs we know the transmission has completed
191  while (NRF_RADIO->EVENTS_DISABLED == 0U)
192  {
193  YIELD;
194  }
195  setModeIdle();
196 
197  return true;

222 // Check whether the latest received message is complete and uncorrupted
223 void RH_NRF51::validateRxBuf()
224 {
225  if (_buf[0] < 4)
226  return; // Too short to be a real message
227  // Extract the 4 headers
228  _rxHeaderTo = _buf[1];
229  _rxHeaderFrom = _buf[2];
230  _rxHeaderId = _buf[3];
231  _rxHeaderFlags = _buf[4];
232  if (_promiscuous ||
233  _rxHeaderTo == _thisAddress ||
234  _rxHeaderTo == RH_BROADCAST_ADDRESS)
235  {
236  _rxGood++;
237  _rxBufValid = true;
238  }

241 bool RH_NRF51::available()
242 {
243  if (!_rxBufValid)
244  {
245  if (_mode == RHModeTx)
246  return false;
247  setModeRx();
248  if (NRF_RADIO->EVENTS_DISABLED == 0U)
249  return false; // No message yet
250  if (NRF_RADIO->CRCSTATUS == ((RADIO_CRCSTATUS_CRCSTATUS_CRCError << RADIO_CRCSTATUS_CRCSTATUS_Pos) & RADIO_CRCSTATUS_CRCSTATUS_Msk))
251  {
252  // Bad CRC, restart the radio
253  _rxBad++;
254  setModeRx();
255  return false;
256  }
257  validateRxBuf();
258  if (_rxBufValid)
259  setModeIdle(); // Got one
260  }
261  return _rxBufValid;
262 }

270 bool RH_NRF51::recv(uint8_t* buf, uint8_t* len)
271 {
272  if (!available())
273  return false;
274  if (buf && len)
275  {
276  // Skip the 4 headers that are at the beginning of the rxBuf
277  // the payload length is the first octet in _buf
278  if (*len > _buf[0]-RH_NRF51_HEADER_LEN)
279  *len = _buf[0]-RH_NRF51_HEADER_LEN;
280  memcpy(buf, _buf+RH_NRF51_HEADER_LEN+1, *len);
281  }
282  clearRxBuf(); // This message accepted and cleared
283  return true;
284 }
