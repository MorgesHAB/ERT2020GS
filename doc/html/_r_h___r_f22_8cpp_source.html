78 bool RH_RF22::init()
79 {
80 #if (RH_PLATFORM == RH_PLATFORM_RPI)
81  if (!RHLinuxSPI::init("/dev/spidev0.1"))
82 #else
83  if (!RHSPIDriver::init())
84 #endif
85  return false;
86 
87  // Determine the interrupt number that corresponds to the interruptPin
88 #if (RH_PLATFORM != RH_PLATFORM_RPI)
89  int interruptNumber = digitalPinToInterrupt(_interruptPin);
90  if (interruptNumber == NOT_AN_INTERRUPT)
91  return false;
92 #endif
93 
94  // Software reset the device
95  reset();
96 
97  // Get the device type and check it
98  // This also tests whether we are really connected to a device
99  _deviceType = spiRead(RH_RF22_REG_00_DEVICE_TYPE);
100  if ( _deviceType != RH_RF22_DEVICE_TYPE_RX_TRX
101  && _deviceType != RH_RF22_DEVICE_TYPE_TX)
102  {
103 #if (RH_PLATFORM == RH_PLATFORM_RPI)
104  printf("Wrong device type!\n");
105 #endif
106  return false;
107  }
108 
109  // Add by Adrien van den Bossche <vandenbo@univ-tlse2.fr> for Teensy
110  // ARM M4 requires the below. else pin interrupt doesn't work properly.
111  // On all other platforms, its innocuous, belt and braces
112 #if (RH_PLATFORM != RH_PLATFORM_RPI)
113  pinMode(_interruptPin, INPUT);
114 #endif
115 
116  // Enable interrupt output on the radio. Interrupt line will now go high until
117  // an interrupt occurs
118  spiWrite(RH_RF22_REG_05_INTERRUPT_ENABLE1, RH_RF22_ENTXFFAEM | RH_RF22_ENRXFFAFULL | RH_RF22_ENPKSENT | RH_RF22_ENPKVALID | RH_RF22_ENCRCERROR | RH_RF22_ENFFERR);
119  spiWrite(RH_RF22_REG_06_INTERRUPT_ENABLE2, RH_RF22_ENPREAVAL);
120 
121  // Set up interrupt handler
122  // Since there are a limited number of interrupt glue functions isr*() available,
123  // we can only support a limited number of devices simultaneously
124  // On some devices, notably most Arduinos, the interrupt pin passed in is actually the
125  // interrupt number. You have to figure out the interruptnumber-to-interruptpin mapping
126  // yourself based on knowledge of what Arduino board you are running on.
127  if (_myInterruptIndex == 0xff)
128  {
129  // First run, no interrupt allocated yet
130  if (_interruptCount <= RH_RF22_NUM_INTERRUPTS)
131  _myInterruptIndex = _interruptCount++;
132  else
133  return false; // Too many devices, not enough interrupt vectors
134  }
135  _deviceForInterrupt[_myInterruptIndex] = this;
136 #if (RH_PLATFORM == RH_PLATFORM_RPI)
137  switch( _myInterruptIndex ){
138  case 0:
139  attachInterrupt(_interruptPin, isr0, INT_EDGE_FALLING);
140  break;
141  case 1:
142  attachInterrupt(_interruptPin, isr1, INT_EDGE_FALLING);
143  break;
144  case 2:
145  attachInterrupt(_interruptPin, isr2, INT_EDGE_FALLING);
146  break;
147  default:
148  return false;
149  }
150 #else
151  if (_myInterruptIndex == 0)
152  attachInterrupt(interruptNumber, isr0, FALLING);
153  else if (_myInterruptIndex == 1)
154  attachInterrupt(interruptNumber, isr1, FALLING);
155  else if (_myInterruptIndex == 2)
156  attachInterrupt(interruptNumber, isr2, FALLING);
157  else
158  return false; // Too many devices, not enough interrupt vectors
159 #endif
160  setModeIdle();
161 
162  clearTxBuf();
163  clearRxBuf();
164 
165  // Most of these are the POR default
166  spiWrite(RH_RF22_REG_7D_TX_FIFO_CONTROL2, RH_RF22_TXFFAEM_THRESHOLD);
167  spiWrite(RH_RF22_REG_7E_RX_FIFO_CONTROL, RH_RF22_RXFFAFULL_THRESHOLD);
168  spiWrite(RH_RF22_REG_30_DATA_ACCESS_CONTROL, RH_RF22_ENPACRX | RH_RF22_ENPACTX | RH_RF22_ENCRC | (_polynomial & RH_RF22_CRC));
169 
170  // Configure the message headers
171  // Here we set up the standard packet format for use by the RH_RF22 library
172  // 8 nibbles preamble
173  // 2 SYNC words 2d, d4
174  // Header length 4 (to, from, id, flags)
175  // 1 octet of data length (0 to 255)
176  // 0 to 255 octets data
177  // 2 CRC octets as CRC16(IBM), computed on the header, length and data
178  // On reception the to address is check for validity against RH_RF22_REG_3F_CHECK_HEADER3
179  // or the broadcast address of 0xff
180  // If no changes are made after this, the transmitted
181  // to address will be 0xff, the from address will be 0xff
182  // and all such messages will be accepted. This permits the out-of the box
183  // RH_RF22 config to act as an unaddresed, unreliable datagram service
184  spiWrite(RH_RF22_REG_32_HEADER_CONTROL1, RH_RF22_BCEN_HEADER3 | RH_RF22_HDCH_HEADER3);
185  spiWrite(RH_RF22_REG_33_HEADER_CONTROL2, RH_RF22_HDLEN_4 | RH_RF22_SYNCLEN_2);
186 
187  setPreambleLength(8);
188  uint8_t syncwords[] = { 0x2d, 0xd4 };
189  setSyncWords(syncwords, sizeof(syncwords));
190  setPromiscuous(false);
191 
192  // Set some defaults. An innocuous ISM frequency, and reasonable pull-in
193  setFrequency(434.0, 0.05);
194 // setFrequency(900.0);
195  // Some slow, reliable default speed and modulation
196  setModemConfig(FSK_Rb2_4Fd36);
197 // setModemConfig(FSK_Rb125Fd125);
198  setGpioReversed(false);
199  // Lowish power
200  setTxPower(RH_RF22_TXPOW_8DBM);
201 
202  return true;
203 }

205 // C++ level interrupt handler for this instance
206 void RH_RF22::handleInterrupt()
207 {
208  uint8_t _lastInterruptFlags[2];
209  // Read the interrupt flags which clears the interrupt
210  spiBurstRead(RH_RF22_REG_03_INTERRUPT_STATUS1, _lastInterruptFlags, 2);
211 
212 #if 0
213  // DEVELOPER TESTING ONLY
214  // Caution: Serial printing in this interrupt routine can cause mysterious crashes
215  Serial.print("interrupt ");
216  Serial.print(_lastInterruptFlags[0], HEX);
217  Serial.print(" ");
218  Serial.println(_lastInterruptFlags[1], HEX);
219  if (_lastInterruptFlags[0] == 0 && _lastInterruptFlags[1] == 0)
220  Serial.println("FUNNY: no interrupt!");
221 #endif
222 
223 #if 0
224  // DEVELOPER TESTING ONLY
225  // TESTING: fake an RH_RF22_IFFERROR
226  static int counter = 0;
227  if (_lastInterruptFlags[0] & RH_RF22_IPKSENT && counter++ == 10)
228  {
229  _lastInterruptFlags[0] = RH_RF22_IFFERROR;
230  counter = 0;
231  }
232 #endif
233 
234  if (_lastInterruptFlags[0] & RH_RF22_IFFERROR)
235  {
236  resetFifos(); // Clears the interrupt
237  if (_mode == RHModeTx)
238  restartTransmit();
239  else if (_mode == RHModeRx)
240  clearRxBuf();
241 // Serial.println("IFFERROR");
242  }
243  // Caution, any delay here may cause a FF underflow or overflow
244  if (_lastInterruptFlags[0] & RH_RF22_ITXFFAEM)
245  {
246  // See if more data has to be loaded into the Tx FIFO
247  sendNextFragment();
248 // Serial.println("ITXFFAEM");
249  }
250  if (_lastInterruptFlags[0] & RH_RF22_IRXFFAFULL)
251  {
252  // Caution, any delay here may cause a FF overflow
253  // Read some data from the Rx FIFO
254  readNextFragment();
255 // Serial.println("IRXFFAFULL");
256  }
257  if (_lastInterruptFlags[0] & RH_RF22_IEXT)
258  {
259  // This is not enabled by the base code, but users may want to enable it
260  handleExternalInterrupt();
261 // Serial.println("IEXT");
262  }
263  if (_lastInterruptFlags[1] & RH_RF22_IWUT)
264  {
265  // This is not enabled by the base code, but users may want to enable it
266  handleWakeupTimerInterrupt();
267 // Serial.println("IWUT");
268  }
269  if (_lastInterruptFlags[0] & RH_RF22_IPKSENT)
270  {
271 // Serial.println("IPKSENT");
272  _txGood++;
273  // Transmission does not automatically clear the tx buffer.
274  // Could retransmit if we wanted
275  // RH_RF22 transitions automatically to Idle
276  _mode = RHModeIdle;
277  }
278  if (_lastInterruptFlags[0] & RH_RF22_IPKVALID)
279  {
280  uint8_t len = spiRead(RH_RF22_REG_4B_RECEIVED_PACKET_LENGTH);
281 // Serial.println("IPKVALID");
282 
283  // May have already read one or more fragments
284  // Get any remaining unread octets, based on the expected length
285  // First make sure we dont overflow the buffer in the case of a stupid length
286  // or partial bad receives
287  if ( len > RH_RF22_MAX_MESSAGE_LEN
288  || len < _bufLen)
289  {
290  _rxBad++;
291  _mode = RHModeIdle;
292  clearRxBuf();
293  return; // Hmmm receiver buffer overflow.
294  }
295 
296  spiBurstRead(RH_RF22_REG_7F_FIFO_ACCESS, _buf + _bufLen, len - _bufLen);
297  _rxHeaderTo = spiRead(RH_RF22_REG_47_RECEIVED_HEADER3);
298  _rxHeaderFrom = spiRead(RH_RF22_REG_48_RECEIVED_HEADER2);
299  _rxHeaderId = spiRead(RH_RF22_REG_49_RECEIVED_HEADER1);
300  _rxHeaderFlags = spiRead(RH_RF22_REG_4A_RECEIVED_HEADER0);
301  _rxGood++;
302  _bufLen = len;
303  _mode = RHModeIdle;
304  _rxBufValid = true;
305  }
306  if (_lastInterruptFlags[0] & RH_RF22_ICRCERROR)
307  {
308 // Serial.println("ICRCERR");
309  _rxBad++;
310  clearRxBuf();
311  resetRxFifo();
312  _mode = RHModeIdle;
313  setModeRx(); // Keep trying
314  }
315  if (_lastInterruptFlags[1] & RH_RF22_IPREAVAL)
316  {
317 // Serial.println("IPREAVAL");
318  _lastRssi = (int8_t)(-120 + ((spiRead(RH_RF22_REG_26_RSSI) / 2)));
319  _lastPreambleTime = millis();
320  resetRxFifo();
321  clearRxBuf();
322  }
323 }

398 // Returns true if centre + (fhch * fhs) is within limits
399 // Caution, different versions of the RH_RF22 support different max freq
400 // so YMMV
401 bool RH_RF22::setFrequency(float centre, float afcPullInRange)
402 {
403  uint8_t fbsel = RH_RF22_SBSEL;
404  uint8_t afclimiter;
405  if (centre < 240.0 || centre > 960.0) // 930.0 for early silicon
406  return false;
407  if (centre >= 480.0)
408  {
409  if (afcPullInRange < 0.0 || afcPullInRange > 0.318750)
410  return false;
411  centre /= 2;
412  fbsel |= RH_RF22_HBSEL;
413  afclimiter = afcPullInRange * 1000000.0 / 1250.0;
414  }
415  else
416  {
417  if (afcPullInRange < 0.0 || afcPullInRange > 0.159375)
418  return false;
419  afclimiter = afcPullInRange * 1000000.0 / 625.0;
420  }
421  centre /= 10.0;
422  float integerPart = floor(centre);
423  float fractionalPart = centre - integerPart;
424 
425  uint8_t fb = (uint8_t)integerPart - 24; // Range 0 to 23
426  fbsel |= fb;
427  uint16_t fc = fractionalPart * 64000;
428  spiWrite(RH_RF22_REG_73_FREQUENCY_OFFSET1, 0); // REVISIT

657 {
