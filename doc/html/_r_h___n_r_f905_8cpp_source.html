16 bool RH_NRF905::init()
17 {
18 #if defined (__MK20DX128__) || defined (__MK20DX256__)
19  // Teensy is unreliable at 8MHz:
20  _spi.setFrequency(RHGenericSPI::Frequency1MHz);
21 #else
22  _spi.setFrequency(RHGenericSPI::Frequency8MHz);
23 #endif
24  if (!RHNRFSPIDriver::init())
25  return false;
26 
27  // Initialise the slave select pin and the tx Enable pin
28  pinMode(_chipEnablePin, OUTPUT);
29  pinMode(_txEnablePin, OUTPUT);
30  digitalWrite(_chipEnablePin, LOW);
31  digitalWrite(_txEnablePin, LOW);
32 
33  // Configure the chip
34  // CRC 16 bits enabled. 16MHz crystal freq
35  spiWriteRegister(RH_NRF905_CONFIG_9, RH_NRF905_CONFIG_9_CRC_EN | RH_NRF905_CONFIG_9_CRC_MODE_16BIT | RH_NRF905_CONFIG_9_XOF_16MHZ);
36 
37  // Make sure we are powered down
38  setModeIdle();
39 
40  // Some innocuous defaults
41  setChannel(108, LOW); // 433.2 MHz
42  setRF(RH_NRF905::TransmitPowerm10dBm);
43 
44  return true;
45 }

74 bool RH_NRF905::setChannel(uint16_t channel, bool hiFrequency)
75 {
76  spiWriteRegister(RH_NRF905_CONFIG_0, channel & RH_NRF905_CONFIG_0_CH_NO);
77  // Set or clear the high bit of the channel
78  uint8_t bit8 = (channel >> 8) & 0x01;
79  uint8_t reg1 = spiReadRegister(RH_NRF905_CONFIG_1);
80  reg1 = (reg1 & ~0x01) | bit8;
81  // Set or clear the HFREQ_PLL bit
82  reg1 &= ~RH_NRF905_CONFIG_1_HFREQ_PLL;
83  if (hiFrequency)
84  reg1 |= RH_NRF905_CONFIG_1_HFREQ_PLL;
85  spiWriteRegister(RH_NRF905_CONFIG_1, reg1);
86  return true;
87 }

138 bool RH_NRF905::send(const uint8_t* data, uint8_t len)
139 {
140  if (len > RH_NRF905_MAX_MESSAGE_LEN)
141  return false;
142  // Set up the headers
143  _buf[0] = _txHeaderTo;
144  _buf[1] = _txHeaderFrom;
145  _buf[2] = _txHeaderId;
146  _buf[3] = _txHeaderFlags;
147  _buf[4] = len;
148  memcpy(_buf+RH_NRF905_HEADER_LEN, data, len);
149  spiBurstWrite(RH_NRF905_REG_W_TX_PAYLOAD, _buf, len + RH_NRF905_HEADER_LEN);
150  setModeTx();
151  // Radio will return to Standby mode after transmission is complete
152  _txGood++;
153  return true;
154 }

196 // Check whether the latest received message is complete and uncorrupted
197 void RH_NRF905::validateRxBuf()
198 {
199  // Check the length
200  uint8_t len = _buf[4];
201  if (len > RH_NRF905_MAX_MESSAGE_LEN)
202  return; // Silly LEN header
203 
204  // Extract the 4 headers
205  _rxHeaderTo = _buf[0];
206  _rxHeaderFrom = _buf[1];
207  _rxHeaderId = _buf[2];
208  _rxHeaderFlags = _buf[3];
209  if (_promiscuous ||
210  _rxHeaderTo == _thisAddress ||
211  _rxHeaderTo == RH_BROADCAST_ADDRESS)
212  {
213  _rxGood++;
214  _bufLen = len + RH_NRF905_HEADER_LEN; // _buf still includes the headers
215  _rxBufValid = true;
216  }
217 }

219 bool RH_NRF905::available()
220 {
221  if (!_rxBufValid)
222  {
223  if (_mode == RHModeTx)
224  return false;
225  setModeRx();
226  if (!(statusRead() & RH_NRF905_STATUS_DR))
227  return false;
228  // Get the message into the RX buffer, so we can inspect the headers
229  // we still dont know how long is the user message
230  spiBurstRead(RH_NRF905_REG_R_RX_PAYLOAD, _buf, RH_NRF905_MAX_PAYLOAD_LEN);
231  validateRxBuf();
232  if (_rxBufValid)
233  setModeIdle(); // Got one
234  }
235  return _rxBufValid;
236 }

244 bool RH_NRF905::recv(uint8_t* buf, uint8_t* len)
245 {
246  if (!available())
247  return false;
248  if (buf && len)
249  {
250  // Skip the 4 headers that are at the beginning of the rxBuf
251  if (*len > _bufLen-RH_NRF905_HEADER_LEN)
252  *len = _bufLen-RH_NRF905_HEADER_LEN;
253  memcpy(buf, _buf+RH_NRF905_HEADER_LEN, *len);
254  }
255  clearRxBuf(); // This message accepted and cleared
256  return true;
257 }
