29 // Discovers a route to the destination (if necessary), sends and
30 // waits for delivery to the next hop (but not for delivery to the final destination)
31 uint8_t RHMesh::sendtoWait(uint8_t* buf, uint8_t len, uint8_t address, uint8_t flags)
32 {
33  if (len > RH_MESH_MAX_MESSAGE_LEN)
34  return RH_ROUTER_ERROR_INVALID_LENGTH;
35 
36  if (address != RH_BROADCAST_ADDRESS)
37  {
38  RoutingTableEntry* route = getRouteTo(address);
39  if (!route && !doArp(address))
40  return RH_ROUTER_ERROR_NO_ROUTE;
41  }
42 
43  // Now have a route. Contruct an application layer message and send it via that route
44  MeshApplicationMessage* a = (MeshApplicationMessage*)&_tmpMessage;
45  a->header.msgType = RH_MESH_MESSAGE_TYPE_APPLICATION;
46  memcpy(a->data, buf, len);
47  return RHRouter::sendtoWait(_tmpMessage, sizeof(RHMesh::MeshMessageHeader) + len, address, flags);
48 }

51 bool RHMesh::doArp(uint8_t address)
52 {
53  // Need to discover a route
54  // Broadcast a route discovery message with nothing in it
55  MeshRouteDiscoveryMessage* p = (MeshRouteDiscoveryMessage*)&_tmpMessage;
56  p->header.msgType = RH_MESH_MESSAGE_TYPE_ROUTE_DISCOVERY_REQUEST;
57  p->destlen = 1;
58  p->dest = address; // Who we are looking for
59  uint8_t error = RHRouter::sendtoWait((uint8_t*)p, sizeof(RHMesh::MeshMessageHeader) + 2, RH_BROADCAST_ADDRESS);
60  if (error != RH_ROUTER_ERROR_NONE)
61  return false;
62 
63  // Wait for a reply, which will be unicast back to us
64  // It will contain the complete route to the destination
65  uint8_t messageLen = sizeof(_tmpMessage);
66  // FIXME: timeout should be configurable
67  unsigned long starttime = millis();
68  int32_t timeLeft;
69  while ((timeLeft = RH_MESH_ARP_TIMEOUT - (millis() - starttime)) > 0)
70  {
71  if (waitAvailableTimeout(timeLeft))
72  {
73  if (RHRouter::recvfromAck(_tmpMessage, &messageLen))
74  {
75  if ( messageLen > 1
76  && p->header.msgType == RH_MESH_MESSAGE_TYPE_ROUTE_DISCOVERY_RESPONSE)
77  {
78  // Got a reply, now add the next hop to the dest to the routing table
79  // The first hop taken is the first octet
80  addRouteTo(address, headerFrom());
81  return true;
82  }
83  }
84  }
85  YIELD;
86  }
87  return false;
88 }

91 // Called by RHRouter::recvfromAck whenever a message goes past
92 void RHMesh::peekAtMessage(RoutedMessage* message, uint8_t messageLen)
93 {
94  MeshMessageHeader* m = (MeshMessageHeader*)message->data;
95  if ( messageLen > 1
96  && m->msgType == RH_MESH_MESSAGE_TYPE_ROUTE_DISCOVERY_RESPONSE)
97  {
98  // This is a unicast RH_MESH_MESSAGE_TYPE_ROUTE_DISCOVERY_RESPONSE messages
99  // being routed back to the originator here. Want to scrape some routing data out of the response
100  // We can find the routes to all the nodes between here and the responding node
101  MeshRouteDiscoveryMessage* d = (MeshRouteDiscoveryMessage*)message->data;
102  addRouteTo(d->dest, headerFrom());
103  uint8_t numRoutes = messageLen - sizeof(RoutedMessageHeader) - sizeof(MeshMessageHeader) - 2;
104  uint8_t i;
105  // Find us in the list of nodes that were traversed to get to the responding node
106  for (i = 0; i < numRoutes; i++)
107  if (d->route[i] == _thisAddress)
108  break;
109  i++;
110  while (i++ < numRoutes)
111  addRouteTo(d->route[i], headerFrom());
112  }
113  else if ( messageLen > 1
114  && m->msgType == RH_MESH_MESSAGE_TYPE_ROUTE_FAILURE)
115  {
116  MeshRouteFailureMessage* d = (MeshRouteFailureMessage*)message->data;
117  deleteRouteTo(d->dest);
118  }
119 }

122 // This is called when a message is to be delivered to the next hop
123 uint8_t RHMesh::route(RoutedMessage* message, uint8_t messageLen)
124 {
125  uint8_t from = headerFrom(); // Might get clobbered during call to superclass route()
126  uint8_t ret = RHRouter::route(message, messageLen);
127  if ( ret == RH_ROUTER_ERROR_NO_ROUTE
128  || ret == RH_ROUTER_ERROR_UNABLE_TO_DELIVER)
129  {
130  // Cant deliver to the next hop. Delete the route
131  deleteRouteTo(message->header.dest);
132  if (message->header.source != _thisAddress)
133  {
134  // This is being proxied, so tell the originator about it
135  MeshRouteFailureMessage* p = (MeshRouteFailureMessage*)&_tmpMessage;
136  p->header.msgType = RH_MESH_MESSAGE_TYPE_ROUTE_FAILURE;
137  p->dest = message->header.dest; // Who you were trying to deliver to
138  // Make sure there is a route back towards whoever sent the original message
139  addRouteTo(message->header.source, from);
140  ret = RHRouter::sendtoWait((uint8_t*)p, sizeof(RHMesh::MeshMessageHeader) + 1, message->header.source);
141  }
142  }
143  return ret;
144 }

155 bool RHMesh::recvfromAck(uint8_t* buf, uint8_t* len, uint8_t* source, uint8_t* dest, uint8_t* id, uint8_t* flags)
156 {
157  uint8_t tmpMessageLen = sizeof(_tmpMessage);
158  uint8_t _source;
159  uint8_t _dest;
160  uint8_t _id;
161  uint8_t _flags;
162  if (RHRouter::recvfromAck(_tmpMessage, &tmpMessageLen, &_source, &_dest, &_id, &_flags))
163  {
164  MeshMessageHeader* p = (MeshMessageHeader*)&_tmpMessage;
165 
166  if ( tmpMessageLen >= 1
167  && p->msgType == RH_MESH_MESSAGE_TYPE_APPLICATION)
168  {
169  MeshApplicationMessage* a = (MeshApplicationMessage*)p;
170  // Handle application layer messages, presumably for our caller
171  if (source) *source = _source;
172  if (dest) *dest = _dest;
173  if (id) *id = _id;
174  if (flags) *flags = _flags;
175  uint8_t msgLen = tmpMessageLen - sizeof(MeshMessageHeader);
176  if (*len > msgLen)
177  *len = msgLen;
178  memcpy(buf, a->data, *len);
179 
180  return true;
181  }
182  else if ( _dest == RH_BROADCAST_ADDRESS
183  && tmpMessageLen > 1
184  && p->msgType == RH_MESH_MESSAGE_TYPE_ROUTE_DISCOVERY_REQUEST)
185  {
186  MeshRouteDiscoveryMessage* d = (MeshRouteDiscoveryMessage*)p;
187  // Handle Route discovery requests
188  // Message is an array of node addresses the route request has already passed through
189  // If it originally came from us, ignore it
190  if (_source == _thisAddress)
191  return false;
192 
193  uint8_t numRoutes = tmpMessageLen - sizeof(MeshMessageHeader) - 2;
194  uint8_t i;
195  // Are we already mentioned?
196  for (i = 0; i < numRoutes; i++)
197  if (d->route[i] == _thisAddress)
198  return false; // Already been through us. Discard
199 
200  // Hasnt been past us yet, record routes back to the earlier nodes
201  addRouteTo(_source, headerFrom()); // The originator
202  for (i = 0; i < numRoutes; i++)
203  addRouteTo(d->route[i], headerFrom());
204  if (isPhysicalAddress(&d->dest, d->destlen))
205  {
206  // This route discovery is for us. Unicast the whole route back to the originator
207  // as a RH_MESH_MESSAGE_TYPE_ROUTE_DISCOVERY_RESPONSE
208  // We are certain to have a route there, because we just got it
209  d->header.msgType = RH_MESH_MESSAGE_TYPE_ROUTE_DISCOVERY_RESPONSE;
210  RHRouter::sendtoWait((uint8_t*)d, tmpMessageLen, _source);
211  }
212  else if (i < _max_hops)
213  {
214  // Its for someone else, rebroadcast it, after adding ourselves to the list
215  d->route[numRoutes] = _thisAddress;
216  tmpMessageLen++;
217  // Have to impersonate the source
218  // REVISIT: if this fails what can we do?
219  RHRouter::sendtoFromSourceWait(_tmpMessage, tmpMessageLen, RH_BROADCAST_ADDRESS, _source);
220  }
221  }
222  }
223  return false;
224 }

227 bool RHMesh::recvfromAckTimeout(uint8_t* buf, uint8_t* len, uint16_t timeout, uint8_t* from, uint8_t* to, uint8_t* id, uint8_t* flags)
228 {

230  int32_t timeLeft;
231  while ((timeLeft = timeout - (millis() - starttime)) > 0)
232  {
233  if (waitAvailableTimeout(timeLeft))
234  {
235  if (recvfromAck(buf, len, from, to, id, flags))
236  return true;
237  YIELD;
238  }
239  }
