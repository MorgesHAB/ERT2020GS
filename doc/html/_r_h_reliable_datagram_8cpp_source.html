47 bool RHReliableDatagram::sendtoWait(uint8_t* buf, uint8_t len, uint8_t address)
48 {
49  // Assemble the message
50  uint8_t thisSequenceNumber = ++_lastSequenceNumber;
51  uint8_t retries = 0;
52  while (retries++ <= _retries)
53  {
54  setHeaderId(thisSequenceNumber);
55  setHeaderFlags(RH_FLAGS_NONE, RH_FLAGS_ACK); // Clear the ACK flag
56  sendto(buf, len, address);
57  waitPacketSent();
58 
59  // Never wait for ACKS to broadcasts:
60  if (address == RH_BROADCAST_ADDRESS)
61  return true;
62 
63  if (retries > 1)
64  _retransmissions++;
65  unsigned long thisSendTime = millis(); // Timeout does not include original transmit time
66 
67  // Compute a new timeout, random between _timeout and _timeout*2
68  // This is to prevent collisions on every retransmit
69  // if 2 nodes try to transmit at the same time
70 #if (RH_PLATFORM == RH_PLATFORM_RASPI) // use standard library random(), bugs in random(min, max)
71  uint16_t timeout = _timeout + (_timeout * (random() & 0xFF) / 256);
72 #else
73  uint16_t timeout = _timeout + (_timeout * random(0, 256) / 256);
74 #endif
75  int32_t timeLeft;
76  while ((timeLeft = timeout - (millis() - thisSendTime)) > 0)
77  {
78  if (waitAvailableTimeout(timeLeft))
79  {
80  uint8_t from, to, id, flags;
81  if (recvfrom(0, 0, &from, &to, &id, &flags)) // Discards the message
82  {
83  // Now have a message: is it our ACK?
84  if ( from == address
85  && to == _thisAddress
86  && (flags & RH_FLAGS_ACK)
87  && (id == thisSequenceNumber))
88  {
89  // Its the ACK we are waiting for
90  return true;
91  }
92  else if ( !(flags & RH_FLAGS_ACK)
93  && (id == _seenIds[from]))
94  {
95  // This is a request we have already received. ACK it again
96  acknowledge(id, from);
97  }
98  // Else discard it
99  }
100  }
101  // Not the one we are waiting for, maybe keep waiting until timeout exhausted
102  YIELD;
103  }
104  // Timeout exhausted, maybe retry
105  YIELD;
106  }
107  // Retries exhausted
108  return false;
109 }

112 bool RHReliableDatagram::recvfromAck(uint8_t* buf, uint8_t* len, uint8_t* from, uint8_t* to, uint8_t* id, uint8_t* flags)
113 {
114  uint8_t _from;
115  uint8_t _to;
116  uint8_t _id;
117  uint8_t _flags;
118  // Get the message before its clobbered by the ACK (shared rx and tx buffer in some drivers
119  if (available() && recvfrom(buf, len, &_from, &_to, &_id, &_flags))
120  {
121  // Never ACK an ACK
122  if (!(_flags & RH_FLAGS_ACK))
123  {
124  // Its a normal message for this node, not an ACK
125  if (_to != RH_BROADCAST_ADDRESS)
126  {
127  // Its not a broadcast, so ACK it
128  // Acknowledge message with ACK set in flags and ID set to received ID
129  acknowledge(_id, _from);
130  }
131  // If we have not seen this message before, then we are interested in it
132  if (_id != _seenIds[_from])
133  {
134  if (from) *from = _from;
135  if (to) *to = _to;
136  if (id) *id = _id;
137  if (flags) *flags = _flags;
138  _seenIds[_from] = _id;
139  return true;
140  }
141  // Else just re-ack it and wait for a new one
142  }
143  }
144  // No message for us available
145  return false;
146 }

174 void RHReliableDatagram::acknowledge(uint8_t id, uint8_t from)
175 {
176  setHeaderId(id);
177  setHeaderFlags(RH_FLAGS_ACK);
178  // We would prefer to send a zero length ACK,
179  // but if an RH_RF22 receives a 0 length message with a CRC error, it will never receive
180  // a 0 length message again, until its reset, which makes everything hang :-(
181  // So we send an ACK of 1 octet
182  // REVISIT: should we send the RSSI for the information of the sender?
183  uint8_t ack = '!';
184  sendto(&ack, sizeof(ack), from);
185  waitPacketSent();
186 }
