109 int RHLinuxSPI::exchange(uint32_t length)
110 {
111  int ret;
112  struct spi_ioc_transfer tr = {0};
113 
114  tr.tx_buf = (unsigned long)_tx_buffer; //tx and rx MUST be the same length!
115  tr.rx_buf = (unsigned long)_rx_buffer;
116  tr.len = length;
117  tr.delay_usecs = 0;
118  tr.speed_hz = _speed;
119  tr.bits_per_word = _bits;
120  tr.pad=0;
121 
122  if( (ret = ioctl(_fd, SPI_IOC_MESSAGE(1), &tr)) < 0)
123  perror("RHLinuxSPI::exchange() : failed");
124 #ifdef RH_LINUX_SPI_DEBUG
125  printf("RHLinuxSPI::exchange() : length=%d ret=%d\n", length, ret);
126 #endif
127 
128  return ret;
129 }
