105 bool RH_RF69::init()
106 {
107 #if (RH_PLATFORM == RH_PLATFORM_RPI)
108  if(!RHLinuxSPI::init("/dev/spidev0.1"))
109 #else
110  if (!RHSPIDriver::init())
111 #endif
112  return false;
113 
114  // Determine the interrupt number that corresponds to the interruptPin
115 #if (RH_PLATFORM != RH_PLATFORM_RPI)
116  int interruptNumber = digitalPinToInterrupt(_interruptPin);
117  if (interruptNumber == NOT_AN_INTERRUPT)
118  return false;
119 #endif
120 
121  // Get the device type and check it
122  // This also tests whether we are really connected to a device
123  // My test devices return 0x24
124  _deviceType = spiRead(RH_RF69_REG_10_VERSION);
125  if (_deviceType == 00 ||
126  _deviceType == 0xff)
127  return false;
128 
129  // Add by Adrien van den Bossche <vandenbo@univ-tlse2.fr> for Teensy
130  // ARM M4 requires the below. else pin interrupt doesn't work properly.
131  // On all other platforms, its innocuous, belt and braces
132 #if (RH_PLATFORM != RH_PLATFORM_RPI)
133  pinMode(_interruptPin, INPUT);
134 #endif
135 
136  // Set up interrupt handler
137  // Since there are a limited number of interrupt glue functions isr*() available,
138  // we can only support a limited number of devices simultaneously
139  // ON some devices, notably most Arduinos, the interrupt pin passed in is actuallt the
140  // interrupt number. You have to figure out the interruptnumber-to-interruptpin mapping
141  // yourself based on knwledge of what Arduino board you are running on.
142  if (_myInterruptIndex == 0xff)
143  {
144  // First run, no interrupt allocated yet
145  if (_interruptCount <= RH_RF69_NUM_INTERRUPTS)
146  _myInterruptIndex = _interruptCount++;
147  else
148  return false; // Too many devices, not enough interrupt vectors
149  }
150  _deviceForInterrupt[_myInterruptIndex] = this;
151 
152 #if (RH_PLATFORM == RH_PLATFORM_RPI)
153  switch( _myInterruptIndex ){
154  case 0:
155  attachInterrupt(_interruptPin, isr0, INT_EDGE_RISING);
156  break;
157  case 1:
158  attachInterrupt(_interruptPin, isr1, INT_EDGE_RISING);
159  break;
160  case 2:
161  attachInterrupt(_interruptPin, isr2, INT_EDGE_RISING);
162  break;
163  default:
164  return false;
165  }
166 #else
167  if (_myInterruptIndex == 0)
168  attachInterrupt(interruptNumber, isr0, RISING);
169  else if (_myInterruptIndex == 1)
170  attachInterrupt(interruptNumber, isr1, RISING);
171  else if (_myInterruptIndex == 2)
172  attachInterrupt(interruptNumber, isr2, RISING);
173  else
174  return false; // Too many devices, not enough interrupt vectors
175 #endif
176 
177  setModeIdle();
178 
179  // Configure important RH_RF69 registers
180  // Here we set up the standard packet format for use by the RH_RF69 library:
181  // 4 bytes preamble
182  // 2 SYNC words 2d, d4
183  // 2 CRC CCITT octets computed on the header, length and data (this in the modem config data)
184  // 0 to 60 bytes data
185  // RSSI Threshold -114dBm
186  // We dont use the RH_RF69s address filtering: instead we prepend our own headers to the beginning
187  // of the RH_RF69 payload
188  spiWrite(RH_RF69_REG_3C_FIFOTHRESH, RH_RF69_FIFOTHRESH_TXSTARTCONDITION_NOTEMPTY | 0x0f); // thresh 15 is default
189  // RSSITHRESH is default
190 // spiWrite(RH_RF69_REG_29_RSSITHRESH, 220); // -110 dbM
191  // SYNCCONFIG is default. SyncSize is set later by setSyncWords()
192 // spiWrite(RH_RF69_REG_2E_SYNCCONFIG, RH_RF69_SYNCCONFIG_SYNCON); // auto, tolerance 0
193  // PAYLOADLENGTH is default
194 // spiWrite(RH_RF69_REG_38_PAYLOADLENGTH, RH_RF69_FIFO_SIZE); // max size only for RX
195  // PACKETCONFIG 2 is default
196  spiWrite(RH_RF69_REG_6F_TESTDAGC, RH_RF69_TESTDAGC_CONTINUOUSDAGC_IMPROVED_LOWBETAOFF);
197  // If high power boost set previously, disable it
198  spiWrite(RH_RF69_REG_5A_TESTPA1, RH_RF69_TESTPA1_NORMAL);
199  spiWrite(RH_RF69_REG_5C_TESTPA2, RH_RF69_TESTPA2_NORMAL);
200 
201  // The following can be changed later by the user if necessary.
202  // Set up default configuration
203  uint8_t syncwords[] = { 0x2d, 0xd4 };
204  setSyncWords(syncwords, sizeof(syncwords)); // Same as RF22's
205  // Reasonably fast and reliable default speed and modulation
206  setModemConfig(GFSK_Rb250Fd250);
207 
208  // 3 would be sufficient, but this is the same as RF22's
209  setPreambleLength(4);
210  // An innocuous ISM frequency, same as RF22's
211  setFrequency(434.0);
212  // No encryption
213  setEncryptionKey(NULL);
214  // +13dBm, same as power-on default
215  setTxPower(13);
216 
217  return true;
218 }

220 // C++ level interrupt handler for this instance
221 // RH_RF69 is unusual in Mthat it has several interrupt lines, and not a single, combined one.
222 // On Moteino, only one of the several interrupt lines (DI0) from the RH_RF69 is connnected to the processor.
223 // We use this to get PACKETSDENT and PAYLOADRADY interrupts.
224 void RH_RF69::handleInterrupt()
225 {
226  // Get the interrupt cause
227  uint8_t irqflags2 = spiRead(RH_RF69_REG_28_IRQFLAGS2);
228  if (_mode == RHModeTx && (irqflags2 & RH_RF69_IRQFLAGS2_PACKETSENT))
229  {
230  // A transmitter message has been fully sent
231  setModeIdle(); // Clears FIFO
232  _txGood++;
233 // Serial.println("PACKETSENT");
234  }
235  // Must look for PAYLOADREADY, not CRCOK, since only PAYLOADREADY occurs _after_ AES decryption
236  // has been done
237  if (_mode == RHModeRx && (irqflags2 & RH_RF69_IRQFLAGS2_PAYLOADREADY))
238  {
239  // A complete message has been received with good CRC
240  _lastRssi = -((int8_t)(spiRead(RH_RF69_REG_24_RSSIVALUE) >> 1));
241  _lastPreambleTime = millis();
242 
243  setModeIdle();
244  // Save it in our buffer
245  readFifo();
246 // Serial.println("PAYLOADREADY");
247  }
248 }
249 
250 // Low level function reads the FIFO and checks the address
251 // Caution: since we put our headers in what the RH_RF69 considers to be the payload, if encryption is enabled
252 // we have to suffer the cost of decryption before we can determine whether the address is acceptable.
253 // Performance issue?
254 void RH_RF69::readFifo()
255 {
256  ATOMIC_BLOCK_START;
257 #if (RH_PLATFORM == RH_PLATFORM_RPI)
258  uint8_t buffer[RH_RF69_MAX_ENCRYPTABLE_PAYLOAD_LEN +1];
259  uint8_t* pBuffer = buffer;
260 
261  uint8_t payloadlen = spiRead(RH_RF69_REG_00_FIFO);
262  if( payloadlen <= RH_RF69_MAX_ENCRYPTABLE_PAYLOAD_LEN &&
263  payloadlen >= RH_RF69_HEADER_LEN ) {
264  spiBurstRead(RH_RF69_REG_00_FIFO, buffer, payloadlen);
265  _rxHeaderTo = *pBuffer++;
266  // Check addressing
267  if (_promiscuous || _rxHeaderTo == _thisAddress ||
268  _rxHeaderTo == RH_BROADCAST_ADDRESS ) {
269  // Get the rest of the headers
270  _rxHeaderFrom = *pBuffer++;
271  _rxHeaderId = *pBuffer++;
272  _rxHeaderFlags = *pBuffer++;
273  _bufLen = payloadlen - RH_RF69_HEADER_LEN;
274  memcpy(_buf, pBuffer, _bufLen);
275  _rxGood++;
276  _rxBufValid = true;
277  }
278  }
279 #else
280  digitalWrite(_slaveSelectPin, LOW);
281  _spi.transfer(RH_RF69_REG_00_FIFO); // Send the start address with the write mask off
282  uint8_t payloadlen = _spi.transfer(0); // First byte is payload len (counting the headers)
283  if (payloadlen <= RH_RF69_MAX_ENCRYPTABLE_PAYLOAD_LEN &&
284  payloadlen >= RH_RF69_HEADER_LEN)
285  {
286  _rxHeaderTo = _spi.transfer(0);
287  // Check addressing
288  if (_promiscuous ||
289  _rxHeaderTo == _thisAddress ||
290  _rxHeaderTo == RH_BROADCAST_ADDRESS)
291  {
292  // Get the rest of the headers
293  _rxHeaderFrom = _spi.transfer(0);
294  _rxHeaderId = _spi.transfer(0);
295  _rxHeaderFlags = _spi.transfer(0);
296  // And now the real payload
297  for (_bufLen = 0; _bufLen < (payloadlen - RH_RF69_HEADER_LEN); _bufLen++)
298  _buf[_bufLen] = _spi.transfer(0);
299  _rxGood++;
300  _rxBufValid = true;
301  }
302  }
303  digitalWrite(_slaveSelectPin, HIGH);
304 #endif
305  ATOMIC_BLOCK_END;
306  // Any junk remaining in the FIFO will be cleared next time we go to receive mode.
307 }

363 {

365  opmode &= ~RH_RF69_OPMODE_MODE;

432 void RH_RF69::setTxPower(int8_t power)
433 {
434  _power = power;
435 
436  uint8_t palevel;
437  if (_power < -18)
438  _power = -18;
439 
440  // See http://www.hoperf.com/upload/rfchip/RF69-V1.2.pdf section 3.3.6
441  // for power formulas
442  if (_power <= 13)
443  {
444  // -18dBm to +13dBm
445  palevel = RH_RF69_PALEVEL_PA0ON | ((_power + 18) & RH_RF69_PALEVEL_OUTPUTPOWER);
446  }
447  else if (_power >= 18)
448  {
449  // +18dBm to +20dBm
450  // Need PA1+PA2
451  // Also need PA boost settings change when tx is turned on and off, see setModeTx()
452  palevel = RH_RF69_PALEVEL_PA1ON | RH_RF69_PALEVEL_PA2ON | ((_power + 11) & RH_RF69_PALEVEL_OUTPUTPOWER);
453  }
454  else
455  {
456  // +14dBm to +17dBm
457  // Need PA1+PA2
458  palevel = RH_RF69_PALEVEL_PA1ON | RH_RF69_PALEVEL_PA2ON | ((_power + 14) & RH_RF69_PALEVEL_OUTPUTPOWER);
459  }
460  spiWrite(RH_RF69_REG_11_PALEVEL, palevel);
461 }

545 bool RH_RF69::send(const uint8_t* data, uint8_t len)
546 {
547  if (len > RH_RF69_MAX_MESSAGE_LEN)
548  return false;
549 
550  waitPacketSent(); // Make sure we dont interrupt an outgoing message
551  setModeIdle(); // Prevent RX while filling the fifo
552 
553 ATOMIC_BLOCK_START;
554 #if (RH_PLATFORM == RH_PLATFORM_RPI)
555  uint8_t buffer[RH_RF69_MAX_MESSAGE_LEN+1];
556  /* Header */
557  txHeader_t *p_TxHeader = (txHeader_t*) buffer;
558  p_TxHeader->length = len + RH_RF69_HEADER_LEN;
559  p_TxHeader->to = _txHeaderTo;
560  p_TxHeader->from = _txHeaderFrom;
561  p_TxHeader->id = _txHeaderId;
562  p_TxHeader->flags = _txHeaderFlags;
563  /* Data */
564  memcpy(&p_TxHeader->data, data, len);
565  /* Sending */
566  spiBurstWrite(RH_RF69_REG_00_FIFO, buffer, len + RH_RF69_HEADER_LEN + 1)
567 #else
568  digitalWrite(_slaveSelectPin, LOW);
569  _spi.transfer(RH_RF69_REG_00_FIFO | RH_RF69_SPI_WRITE_MASK); // Send the start address with the write mask on
570  _spi.transfer(len + RH_RF69_HEADER_LEN); // Include length of headers
571  // First the 4 headers
572  _spi.transfer(_txHeaderTo);
573  _spi.transfer(_txHeaderFrom);
574  _spi.transfer(_txHeaderId);
575  _spi.transfer(_txHeaderFlags);
576  // Now the payload
577  while (len--)
578  _spi.transfer(*data++);
579  digitalWrite(_slaveSelectPin, HIGH);
580 #endif
581  ATOMIC_BLOCK_END;
582  setModeTx(); // Start the transmitter
583 
584  return true;
585 }
