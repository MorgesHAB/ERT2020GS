148 #ifdef B76800
149  else if (baud == 76800) // Not available on Linux

156 #ifdef B460800
157  else if (baud == 460800) // Not available on OSX
158  speed = B460800;
159 #endif
160 #ifdef B921600
161  else if (baud == 921600) // Not available on OSX
162  speed = B921600;
163 #endif
164  else
165  {
166  fprintf(stderr, "HardwareSerial::setBaud: unsupported baud rate %d\n", baud);
167  return false;
168  }
169 
170  struct termios options;
171  // Get current options
172  if (tcgetattr(_device, &options) != 0)
173  {
174  fprintf(stderr, "HardwareSerial::setBaud: could not tcgetattr %s\n", strerror(errno));
175  return false;
176  }
177 
178  // Set new speed options
179  cfsetispeed(&options, speed);
180  cfsetospeed(&options, speed);
181  // Enable the receiver and set local mode...
182  options.c_cflag |= (CLOCAL | CREAD);
183 
184  // Force mode to 8,N,1
185  // to be compatible with Arduino HardwareSerial
186  // Should this be configurable? Prob not, must have 8 bits, dont need parity.
187  options.c_cflag &= ~(PARENB | CSTOPB | CSIZE);
188  options.c_cflag |= CS8;
189 
190  // Disable flow control and input character conversions
191  options.c_iflag &= ~(IXON | IXOFF | IXANY | ICRNL | INLCR);
192 
193  // Raw input:
194  options.c_lflag &= ~(ICANON | ECHO | ECHOE | ISIG);
195 
196  // Raw output
197  options.c_oflag &= ~(OPOST | OCRNL | ONLCR);
198 
199  // Set the options in the port
200  if (tcsetattr(_device, TCSANOW, &options) != 0)
201  {
202  fprintf(stderr, "HardwareSerial::setBaud: could not tcsetattr %s\n", strerror(errno));
203  return false;
204  }
205 
206  _baud = baud;
207  return true;
208 }

216 // Block until something is available or timeout expires
217 bool HardwareSerial::waitAvailableTimeout(uint16_t timeout)
218 {
219  int max_fd;
220  fd_set input;
221  int result;
222 
223  FD_ZERO(&input);
224  FD_SET(_device, &input);
225  max_fd = _device + 1;
226 
227  if (timeout)
228  {
229  struct timeval timer;
230  // Timeout is in milliseconds
231  timer.tv_sec = timeout / 1000;
232  timer.tv_usec = (timeout % 1000) * 1000;
233  result = select(max_fd, &input, NULL, NULL, &timer);
234  }
235  else
236  {
237  result = select(max_fd, &input, NULL, NULL, NULL);
238  }
239  if (result < 0)
240  fprintf(stderr, "HardwareSerial::waitAvailableTimeout: select failed %s\n", strerror(errno));
241  return result > 0;
242 }
