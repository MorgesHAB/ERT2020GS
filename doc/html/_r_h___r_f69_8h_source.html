43 // This is the maximum message length that can be supported by this driver. Limited by
44 // the size of the FIFO, since we are unable to support on-the-fly filling and emptying
45 // of the FIFO.
46 // Can be pre-defined to a smaller size (to save SRAM) prior to including this header
47 // Here we allow for 4 bytes of address and header and payload to be included in the 64 byte encryption limit.
48 // the one byte payload length is not encrpyted
49 #ifndef RH_RF69_MAX_MESSAGE_LEN
50 #define RH_RF69_MAX_MESSAGE_LEN (RH_RF69_MAX_ENCRYPTABLE_PAYLOAD_LEN - RH_RF69_HEADER_LEN)
