22 // 4 bit to 6 bit symbol converter table

24 // into 6 bit symbols for transmission. Each 6-bit symbol has 3 1s and 3 0s

91 // Common function for setting timer ticks @ prescaler values for speed
92 // Returns prescaler index into {0, 1, 8, 64, 256, 1024} array
93 // and sets nticks to compare-match value if lower than max_ticks
94 // returns 0 & nticks = 0 on fault
95 uint8_t RH_ASK::timerCalc(uint16_t speed, uint16_t max_ticks, uint16_t *nticks)
96 {
97 #if (RH_PLATFORM == RH_PLATFORM_ARDUINO) || (RH_PLATFORM == RH_PLATFORM_GENERIC_AVR8)
98  // Clock divider (prescaler) values - 0/3333: error flag
99  uint8_t prescaler; // index into array & return bit value
100  unsigned long ulticks; // calculate by ntick overflow
101 
102  // Div-by-zero protection
103  if (speed == 0)
104  {
105  // signal fault
106  *nticks = 0;
107  return 0;
108  }
109 
110  // test increasing prescaler (divisor), decreasing ulticks until no overflow
111  // 1/Fraction of second needed to xmit one bit
112  unsigned long inv_bit_time = ((unsigned long)speed) * 8;
113  for (prescaler=1; prescaler < NUM_PRESCALERS; prescaler += 1)
114  {
115  // Integer arithmetic courtesy Jim Remington
116  // 1/Amount of time per CPU clock tick (in seconds)
117  uint16_t prescalerValue;
118  memcpy_P(&prescalerValue, &prescalers[prescaler], sizeof(uint16_t));
119  unsigned long inv_clock_time = F_CPU / ((unsigned long)prescalerValue);
120  // number of prescaled ticks needed to handle bit time @ speed
121  ulticks = inv_clock_time / inv_bit_time;
122 
123  // Test if ulticks fits in nticks bitwidth (with 1-tick safety margin)
124  if ((ulticks > 1) && (ulticks < max_ticks))
125  break; // found prescaler
126 
127  // Won't fit, check with next prescaler value
128  }
129 
130 
131  // Check for error
132  if ((prescaler == 6) || (ulticks < 2) || (ulticks > max_ticks))
133  {
134  // signal fault
135  *nticks = 0;
136  return 0;
137  }
138 
139  *nticks = ulticks;
140  return prescaler;
141 #else
142  return 0; // not implemented or needed on other platforms
143 #endif
144 }

146 // The idea here is to get 8 timer interrupts per bit period
147 void RH_ASK::timerSetup()
148 {
149 #if (RH_PLATFORM == RH_PLATFORM_GENERIC_AVR8)
150  uint16_t nticks;
151  uint8_t prescaler = timerCalc(_speed, (uint16_t)-1, &nticks);
152  if (!prescaler) return;
153  _COMB(TCCR,RH_ASK_TIMER_INDEX,A)= 0;
154  _COMB(TCCR,RH_ASK_TIMER_INDEX,B)= _BV(WGM12);
155  _COMB(TCCR,RH_ASK_TIMER_INDEX,B)|= prescaler;
156  _COMB(OCR,RH_ASK_TIMER_INDEX,A)= nticks;
157  _COMB(TI,MSK,RH_ASK_TIMER_INDEX)|= _BV(_COMB(OCIE,RH_ASK_TIMER_INDEX,A));
158 
159 #elif (RH_PLATFORM == RH_PLATFORM_MSP430) // LaunchPad specific
160  // Calculate the counter overflow count based on the required bit speed
161  // and CPU clock rate
162  uint16_t ocr1a = (F_CPU / 8UL) / _speed;
163 
164  // This code is for Energia/MSP430
165  TA0CCR0 = ocr1a; // Ticks for 62,5 us
166  TA0CTL = TASSEL_2 + MC_1; // SMCLK, up mode
167  TA0CCTL0 |= CCIE; // CCR0 interrupt enabled
168 
169 #elif (RH_PLATFORM == RH_PLATFORM_ARDUINO) // Arduino specific
170  uint16_t nticks; // number of prescaled ticks needed
171  uint8_t prescaler; // Bit values for CS0[2:0]
172 
173  #ifdef RH_PLATFORM_ATTINY
174  // figure out prescaler value and counter match value
175  // REVISIT: does not correctly handle 1MHz clock speeds, only works with 8MHz clocks
176  // At 1MHz clock, get 1/8 of the expected baud rate
177  prescaler = timerCalc(_speed, (uint8_t)-1, &nticks);
178  if (!prescaler)
179  return; // fault
180 
181  TCCR0A = 0;
182  TCCR0A = _BV(WGM01); // Turn on CTC mode / Output Compare pins disconnected
183 
184  // convert prescaler index to TCCRnB prescaler bits CS00, CS01, CS02
185  TCCR0B = 0;
186  TCCR0B = prescaler; // set CS00, CS01, CS02 (other bits not needed)
187 
188 
189  // Number of ticks to count before firing interrupt
190  OCR0A = uint8_t(nticks);
191 
192  // Set mask to fire interrupt when OCF0A bit is set in TIFR0
193 #ifdef TIMSK0
194  // ATtiny84
195  TIMSK0 |= _BV(OCIE0A);
196 #else
197  // ATtiny85
198  TIMSK |= _BV(OCIE0A);
199 #endif
200 
201 
202  #elif defined(__arm__) && defined(CORE_TEENSY)
203  // on Teensy 3.0 (32 bit ARM), use an interval timer
204  IntervalTimer *t = new IntervalTimer();
205  void TIMER1_COMPA_vect(void);
206  t->begin(TIMER1_COMPA_vect, 125000 / _speed);
207 
208  #elif defined(__arm__)
209  // Arduino Due
210  // Due has 9 timers in 3 blocks of 3.
211  // We use timer 1 TC1_IRQn on TC0 channel 1, since timers 0, 2, 3, 4, 5 are used by the Servo library
212  #define RH_ASK_DUE_TIMER TC0
213  #define RH_ASK_DUE_TIMER_CHANNEL 1
214  #define RH_ASK_DUE_TIMER_IRQ TC1_IRQn
215  pmc_set_writeprotect(false);
216  pmc_enable_periph_clk(RH_ASK_DUE_TIMER_IRQ);
217 
218  // Clock speed 4 can handle all reasonable _speeds we might ask for. Its divisor is 128
219  // and we want 8 interrupts per bit
220  uint32_t rc = (VARIANT_MCK / _speed) / 128 / 8;
221  TC_Configure(RH_ASK_DUE_TIMER, RH_ASK_DUE_TIMER_CHANNEL,
222  TC_CMR_WAVE | TC_CMR_WAVSEL_UP_RC | TC_CMR_TCCLKS_TIMER_CLOCK4);
223  TC_SetRC(RH_ASK_DUE_TIMER, RH_ASK_DUE_TIMER_CHANNEL, rc);
224  // Enable the RC Compare Interrupt
225  RH_ASK_DUE_TIMER->TC_CHANNEL[RH_ASK_DUE_TIMER_CHANNEL].TC_IER = TC_IER_CPCS;
226  NVIC_ClearPendingIRQ(RH_ASK_DUE_TIMER_IRQ);
227  NVIC_EnableIRQ(RH_ASK_DUE_TIMER_IRQ);
228  TC_Start(RH_ASK_DUE_TIMER, RH_ASK_DUE_TIMER_CHANNEL);
229 
230  #else
231  // This is the path for most Arduinos
232  // figure out prescaler value and counter match value
233  #if defined(RH_ASK_ARDUINO_USE_TIMER2)
234  prescaler = timerCalc(_speed, (uint8_t)-1, &nticks);
235  if (!prescaler)
236  return; // fault
237  // Use timer 2
238  TCCR2A = _BV(WGM21); // Turn on CTC mode)
239  // convert prescaler index to TCCRnB prescaler bits CS10, CS11, CS12
240  TCCR2B = prescaler;
241 
242  // Caution: special procedures for setting 16 bit regs
243  // is handled by the compiler
244  OCR2A = nticks;
245  // Enable interrupt
246  #ifdef TIMSK2
247  // atmega168
248  TIMSK2 |= _BV(OCIE2A);
249  #else
250  // others
251  TIMSK |= _BV(OCIE2A);
252  #endif // TIMSK2
253  #else
254  // Use timer 1
255  prescaler = timerCalc(_speed, (uint16_t)-1, &nticks);
256  if (!prescaler)
257  return; // fault
258  TCCR1A = 0; // Output Compare pins disconnected
259  TCCR1B = _BV(WGM12); // Turn on CTC mode
260 
261  // convert prescaler index to TCCRnB prescaler bits CS10, CS11, CS12
262  TCCR1B |= prescaler;
263 
264  // Caution: special procedures for setting 16 bit regs
265  // is handled by the compiler
266  OCR1A = nticks;
267  // Enable interrupt
268  #ifdef TIMSK1
269  // atmega168
270  TIMSK1 |= _BV(OCIE1A);
271  #else
272  // others
273  TIMSK |= _BV(OCIE1A);
274  #endif // TIMSK1
275  #endif
276  #endif
277 
278 #elif (RH_PLATFORM == RH_PLATFORM_STM32) // Maple etc
279  // Pause the timer while we're configuring it
280  timer.pause();
281  timer.setPeriod((1000000/8)/_speed);
282  // Set up an interrupt on channel 1
283  timer.setChannel1Mode(TIMER_OUTPUT_COMPARE);
284  timer.setCompare(TIMER_CH1, 1); // Interrupt 1 count after each update

296  OpenTimer1(T1_ON | T1_PS_1_1 | T1_SOURCE_INT, (F_CPU / 8) / _speed);

354 bool RH_ASK::recv(uint8_t* buf, uint8_t* len)
355 {
356  if (!available())
357  return false;
358 
359  if (buf && len)
360  {
361  // Skip the length and 4 headers that are at the beginning of the rxBuf
362  // and drop the trailing 2 bytes of FCS
363  uint8_t message_len = _rxBufLen-RH_ASK_HEADER_LEN - 3;
364  if (*len > message_len)
365  *len = message_len;
366  memcpy(buf, _rxBuf+RH_ASK_HEADER_LEN+1, *len);
367  }
368  _rxBufValid = false; // Got the most recent message, delete it
369 // printBuffer("recv:", buf, *len);
370  return true;

373 // Caution: this may block
374 bool RH_ASK::send(const uint8_t* data, uint8_t len)
375 {
376  uint8_t i;
377  uint16_t index = 0;
378  uint16_t crc = 0xffff;
379  uint8_t *p = _txBuf + RH_ASK_PREAMBLE_LEN; // start of the message area
380  uint8_t count = len + 3 + RH_ASK_HEADER_LEN; // Added byte count and FCS and headers to get total number of bytes
381 
382  if (len > RH_ASK_MAX_MESSAGE_LEN)
383  return false;
384 
385  // Wait for transmitter to become available
386  waitPacketSent();
387 
388  // Encode the message length
389  crc = RHcrc_ccitt_update(crc, count);
390  p[index++] = symbols[count >> 4];
391  p[index++] = symbols[count & 0xf];
392 
393  // Encode the headers
394  crc = RHcrc_ccitt_update(crc, _txHeaderTo);
395  p[index++] = symbols[_txHeaderTo >> 4];
396  p[index++] = symbols[_txHeaderTo & 0xf];
397  crc = RHcrc_ccitt_update(crc, _txHeaderFrom);
398  p[index++] = symbols[_txHeaderFrom >> 4];
399  p[index++] = symbols[_txHeaderFrom & 0xf];
400  crc = RHcrc_ccitt_update(crc, _txHeaderId);
401  p[index++] = symbols[_txHeaderId >> 4];
402  p[index++] = symbols[_txHeaderId & 0xf];
403  crc = RHcrc_ccitt_update(crc, _txHeaderFlags);
404  p[index++] = symbols[_txHeaderFlags >> 4];
405  p[index++] = symbols[_txHeaderFlags & 0xf];
406 
407  // Encode the message into 6 bit symbols. Each byte is converted into
408  // 2 6-bit symbols, high nybble first, low nybble second
409  for (i = 0; i < len; i++)
410  {
411  crc = RHcrc_ccitt_update(crc, data[i]);
412  p[index++] = symbols[data[i] >> 4];
413  p[index++] = symbols[data[i] & 0xf];
414  }
415 
416  // Append the fcs, 16 bits before encoding (4 6-bit symbols after encoding)
417  // Caution: VW expects the _ones_complement_ of the CCITT CRC-16 as the FCS
418  // VW sends FCS as low byte then hi byte
419  crc = ~crc;
420  p[index++] = symbols[(crc >> 4) & 0xf];
421  p[index++] = symbols[crc & 0xf];
422  p[index++] = symbols[(crc >> 12) & 0xf];
423  p[index++] = symbols[(crc >> 8) & 0xf];
424 
425  // Total number of 6-bit symbols to send
426  _txBufLen = index + RH_ASK_PREAMBLE_LEN;
427 
428  // Start the low level interrupt handler sending symbols
429  setModeTx();
430 
431 // FIXME
432  thisASKDriver = this;
433 
434  return true;
435 }

542 // Convert a 6 bit encoded symbol into its 4 bit decoded equivalent
543 uint8_t RH_ASK::symbol_6to4(uint8_t symbol)
544 {
545  uint8_t i;
546  uint8_t count;
547 
548  // Linear search :-( Could have a 64 byte reverse lookup table?
549  // There is a little speedup here courtesy Ralph Doncaster:
550  // The shortcut works because bit 5 of the symbol is 1 for the last 8
551  // symbols, and it is 0 for the first 8.
552  // So we only have to search half the table
553  for (i = (symbol>>2) & 8, count=8; count-- ; i++)
554  if (symbol == symbols[i]) return i;
555 
556  return 0; // Not found
557 }

559 // Check whether the latest received message is complete and uncorrupted
560 // We should always check the FCS at user level, not interrupt level
561 // since it is slow
562 void RH_ASK::validateRxBuf()
563 {
564  uint16_t crc = 0xffff;
565  // The CRC covers the byte count, headers and user data
566  for (uint8_t i = 0; i < _rxBufLen; i++)
567  crc = RHcrc_ccitt_update(crc, _rxBuf[i]);
568  if (crc != 0xf0b8) // CRC when buffer and expected CRC are CRC'd
569  {
570  // Reject and drop the message
571  _rxBad++;
572  _rxBufValid = false;
573  return;
574  }
575 
576  // Extract the 4 headers that follow the message length
577  _rxHeaderTo = _rxBuf[1];
578  _rxHeaderFrom = _rxBuf[2];
579  _rxHeaderId = _rxBuf[3];
580  _rxHeaderFlags = _rxBuf[4];
581  if (_promiscuous ||
582  _rxHeaderTo == _thisAddress ||
583  _rxHeaderTo == RH_BROADCAST_ADDRESS)
584  {
585  _rxGood++;
586  _rxBufValid = true;
587  }

590 void RH_ASK::receiveTimer()
591 {
592  bool rxSample = readRx();
593 
594  // Integrate each sample
595  if (rxSample)
596  _rxIntegrator++;
597 
598  if (rxSample != _rxLastSample)
599  {
600  // Transition, advance if ramp > 80, retard if < 80
601  _rxPllRamp += ((_rxPllRamp < RH_ASK_RAMP_TRANSITION)
602  ? RH_ASK_RAMP_INC_RETARD
603  : RH_ASK_RAMP_INC_ADVANCE);
604  _rxLastSample = rxSample;
605  }
606  else
607  {
608  // No transition
609  // Advance ramp by standard 20 (== 160/8 samples)
610  _rxPllRamp += RH_ASK_RAMP_INC;
611  }
612  if (_rxPllRamp >= RH_ASK_RX_RAMP_LEN)
613  {
614  // Add this to the 12th bit of _rxBits, LSB first
615  // The last 12 bits are kept
616  _rxBits >>= 1;
617 
618  // Check the integrator to see how many samples in this cycle were high.
619  // If < 5 out of 8, then its declared a 0 bit, else a 1;
620  if (_rxIntegrator >= 5)
621  _rxBits |= 0x800;
622 
623  _rxPllRamp -= RH_ASK_RX_RAMP_LEN;
624  _rxIntegrator = 0; // Clear the integral for the next cycle
625 
626  if (_rxActive)
627  {
628  // We have the start symbol and now we are collecting message bits,
629  // 6 per symbol, each which has to be decoded to 4 bits
630  if (++_rxBitCount >= 12)
631  {
632  // Have 12 bits of encoded message == 1 byte encoded
633  // Decode as 2 lots of 6 bits into 2 lots of 4 bits
634  // The 6 lsbits are the high nybble
635  uint8_t this_byte =
636  (symbol_6to4(_rxBits & 0x3f)) << 4
637  | symbol_6to4(_rxBits >> 6);
638 
639  // The first decoded byte is the byte count of the following message
640  // the count includes the byte count and the 2 trailing FCS bytes
641  // REVISIT: may also include the ACK flag at 0x40
642  if (_rxBufLen == 0)
643  {
644  // The first byte is the byte count
645  // Check it for sensibility. It cant be less than 7, since it
646  // includes the byte count itself, the 4 byte header and the 2 byte FCS
647  _rxCount = this_byte;
648  if (_rxCount < 7 || _rxCount > RH_ASK_MAX_PAYLOAD_LEN)
649  {
650  // Stupid message length, drop the whole thing
651  _rxActive = false;
652  _rxBad++;
653  return;
654  }
655  }
656  _rxBuf[_rxBufLen++] = this_byte;
657 
658  if (_rxBufLen >= _rxCount)
659  {
660  // Got all the bytes now
661  _rxActive = false;
662  _rxBufFull = true;
663  setModeIdle();
664  }
665  _rxBitCount = 0;
666  }
667  }
668  // Not in a message, see if we have a start symbol
669  else if (_rxBits == RH_ASK_START_SYMBOL)
670  {
671  // Have start symbol, start collecting message
672  _rxActive = true;
673  _rxBitCount = 0;
674  _rxBufLen = 0;
675  }
676  }
677 }

679 void RH_ASK::transmitTimer()
680 {
681  if (_txSample++ == 0)
682  {
683  // Send next bit
684  // Symbols are sent LSB first
685  // Finished sending the whole message? (after waiting one bit period
686  // since the last bit)
687  if (_txIndex >= _txBufLen)
688  {
689  setModeIdle();
690  _txGood++;
691  }
692  else
693  {
694  writeTx(_txBuf[_txIndex] & (1 << _txBit++));
695  if (_txBit >= 6)
696  {
697  _txBit = 0;
698  _txIndex++;
699  }
700  }
701  }
702 
703  if (_txSample > 7)
704  _txSample = 0;
705 }
