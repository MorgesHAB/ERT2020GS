67 void RH_Serial::handleRx(uint8_t ch)
68 {
69  // State machine for receiving chars
70  switch(_rxState)
71  {
72  case RxStateIdle:
73  {
74  if (ch == DLE)
75  _rxState = RxStateDLE;
76  }
77  break;
78 
79  case RxStateDLE:
80  {
81  if (ch == STX)
82  {
83  clearRxBuf();
84  _rxState = RxStateData;
85  }
86  else
87  _rxState = RxStateIdle;
88  }
89  break;
90 
91  case RxStateData:
92  {
93  if (ch == DLE)
94  _rxState = RxStateEscape;
95  else
96  appendRxBuf(ch);
97  }
98  break;
99 
100  case RxStateEscape:
101  {
102  if (ch == ETX)
103  {
104  // add fcs for DLE, ETX
105  _rxFcs = RHcrc_ccitt_update(_rxFcs, DLE);
106  _rxFcs = RHcrc_ccitt_update(_rxFcs, ETX);
107  _rxState = RxStateWaitFCS1; // End frame
108  }
109  else if (ch == DLE)
110  {
111  appendRxBuf(ch);
112  _rxState = RxStateData;
113  }
114  else
115  _rxState = RxStateIdle; // Unexpected
116  }
117  break;
118 
119  case RxStateWaitFCS1:
120  {
121  _rxRecdFcs = ch << 8;
122  _rxState = RxStateWaitFCS2;
123  }
124  break;
125 
126  case RxStateWaitFCS2:
127  {
128  _rxRecdFcs |= ch;
129  _rxState = RxStateIdle;
130  validateRxBuf();
131  }
132  break;
133 
134  default: // Else some compilers complain
135  break;
136  }
137 }

146 void RH_Serial::appendRxBuf(uint8_t ch)
147 {
148  if (_rxBufLen < RH_SERIAL_MAX_PAYLOAD_LEN)
149  {
150  // Normal data, save and add to FCS
151  _rxBuf[_rxBufLen++] = ch;
152  _rxFcs = RHcrc_ccitt_update(_rxFcs, ch);
153  }
154  // If the buffer overflows, we dont record the trailing data, and the FCS will be wrong,
155  // causing the message to be dropped when the FCS is received
156 }

158 // Check whether the latest received message is complete and uncorrupted
159 void RH_Serial::validateRxBuf()
160 {
161  if (_rxRecdFcs != _rxFcs)
162  {
163  _rxBad++;
164  return;
165  }
166 
167  // Extract the 4 headers
168  _rxHeaderTo = _rxBuf[0];
169  _rxHeaderFrom = _rxBuf[1];
170  _rxHeaderId = _rxBuf[2];
171  _rxHeaderFlags = _rxBuf[3];
172  if (_promiscuous ||
173  _rxHeaderTo == _thisAddress ||
174  _rxHeaderTo == RH_BROADCAST_ADDRESS)
175  {
176  _rxGood++;
177  _rxBufValid = true;

181 bool RH_Serial::recv(uint8_t* buf, uint8_t* len)
182 {
183  if (!available())
184  return false;
185 
186  if (buf && len)
187  {
188  // Skip the 4 headers that are at the beginning of the rxBuf
189  if (*len > _rxBufLen-RH_SERIAL_HEADER_LEN)
190  *len = _rxBufLen-RH_SERIAL_HEADER_LEN;
191  memcpy(buf, _rxBuf+RH_SERIAL_HEADER_LEN, *len);
192  }
193  clearRxBuf(); // This message accepted and cleared
194  return true;
195 }

198 bool RH_Serial::send(const uint8_t* data, uint8_t len)
199 {
200  _txFcs = 0xffff; // Initial value
201  _serial.write(DLE); // Not in FCS
202  _serial.write(STX); // Not in FCS
203  // First the 4 headers
204  txData(_txHeaderTo);
205  txData(_txHeaderFrom);
206  txData(_txHeaderId);
207  txData(_txHeaderFlags);
208  // Now the payload
209  while (len--)
210  txData(*data++);
211  // End of message
212  _serial.write(DLE);
213  _txFcs = RHcrc_ccitt_update(_txFcs, DLE);
214  _serial.write(ETX);
215  _txFcs = RHcrc_ccitt_update(_txFcs, ETX);
216 
217  // Now send the calculated FCS for this message
218  _serial.write((_txFcs >> 8) & 0xff);
219  _serial.write(_txFcs & 0xff);
220  return true;
221 }
