60 void RHHardwareSPI::begin()
61 {
62  // Sigh: there are no common symbols for some of these SPI options across all platforms
63 #if (RH_PLATFORM == RH_PLATFORM_ARDUINO) || (RH_PLATFORM == RH_PLATFORM_UNO32)
64  uint8_t dataMode;
65  if (_dataMode == DataMode0)
66  dataMode = SPI_MODE0;
67  else if (_dataMode == DataMode1)
68  dataMode = SPI_MODE1;
69  else if (_dataMode == DataMode2)
70  dataMode = SPI_MODE2;
71  else if (_dataMode == DataMode3)
72  dataMode = SPI_MODE3;
73  else
74  dataMode = SPI_MODE0;
75 #if (RH_PLATFORM == RH_PLATFORM_ARDUINO) && defined(__arm__) && defined(CORE_TEENSY)
76  // Temporary work-around due to problem where avr_emulation.h does not work properly for the setDataMode() cal
77  SPCR &= ~SPI_MODE_MASK;
78 #else
79  SPI.setDataMode(dataMode);
80 #endif
81 
82 #if (RH_PLATFORM == RH_PLATFORM_ARDUINO) && defined (__arm__) && !defined(CORE_TEENSY)
83  // Arduino Due in 1.5.5 has its own BitOrder :-(
84  ::BitOrder bitOrder;
85 #else
86  uint8_t bitOrder;
87 #endif
88  if (_bitOrder == BitOrderLSBFirst)
89  bitOrder = LSBFIRST;
90  else
91  bitOrder = MSBFIRST;
92  SPI.setBitOrder(bitOrder);
93 
94  uint8_t divider;
95  switch (_frequency)
96  {
97  case Frequency1MHz:
98  default:
99 #if F_CPU == 8000000
100  divider = SPI_CLOCK_DIV8;
101 #else
102  divider = SPI_CLOCK_DIV16;
103 #endif
104  break;
105 
106  case Frequency2MHz:
107 #if F_CPU == 8000000
108  divider = SPI_CLOCK_DIV4;
109 #else
110  divider = SPI_CLOCK_DIV8;
111 #endif
112  break;
113 
114  case Frequency4MHz:
115 #if F_CPU == 8000000
116  divider = SPI_CLOCK_DIV2;
117 #else
118  divider = SPI_CLOCK_DIV4;
119 #endif
120  break;
121 
122  case Frequency8MHz:
123  divider = SPI_CLOCK_DIV2; // 4MHz on an 8MHz Arduino
124  break;
125 
126  case Frequency16MHz:
127  divider = SPI_CLOCK_DIV2; // Not really 16MHz, only 8MHz. 4MHz on an 8MHz Arduino
128  break;
129 
130  }
131  SPI.setClockDivider(divider);
132  SPI.begin();
133 
134 #elif (RH_PLATFORM == RH_PLATFORM_STM32) // Maple etc
135  spi_mode dataMode;
136  // Hmmm, if we do this as a switch, GCC on maple gets v confused!
137  if (_dataMode == DataMode0)
138  dataMode = SPI_MODE_0;
139  else if (_dataMode == DataMode1)
140  dataMode = SPI_MODE_1;
141  else if (_dataMode == DataMode2)
142  dataMode = SPI_MODE_2;
143  else if (_dataMode == DataMode3)
144  dataMode = SPI_MODE_3;
145  else
146  dataMode = SPI_MODE_0;
147 
148  uint32 bitOrder;
149  if (_bitOrder == BitOrderLSBFirst)
150  bitOrder = LSBFIRST;
151  else
152  bitOrder = MSBFIRST;
153 
154  SPIFrequency frequency; // Yes, I know these are not exact equivalents.
155  switch (_frequency)
156  {
157  case Frequency1MHz:
158  default:
159  frequency = SPI_1_125MHZ;
160  break;
161 
162  case Frequency2MHz:
163  frequency = SPI_2_25MHZ;
164  break;
165 
166  case Frequency4MHz:
167  frequency = SPI_4_5MHZ;
168  break;
169 
170  case Frequency8MHz:
171  frequency = SPI_9MHZ;
172  break;
173 
174  case Frequency16MHz:
175  frequency = SPI_18MHZ;
176  break;
177 
178  }
179  SPI.begin(frequency, bitOrder, dataMode);
180 
181 #elif (RH_PLATFORM == RH_PLATFORM_STM32STD) // STM32F4 discovery
182  uint8_t dataMode;
183  if (_dataMode == DataMode0)
184  dataMode = SPI_MODE0;
185  else if (_dataMode == DataMode1)
186  dataMode = SPI_MODE1;
187  else if (_dataMode == DataMode2)
188  dataMode = SPI_MODE2;
189  else if (_dataMode == DataMode3)
190  dataMode = SPI_MODE3;
191  else
192  dataMode = SPI_MODE0;
193 
194  uint32_t bitOrder;
195  if (_bitOrder == BitOrderLSBFirst)
196  bitOrder = LSBFIRST;
197  else
198  bitOrder = MSBFIRST;
199 
200  SPIFrequency frequency; // Yes, I know these are not exact equivalents.
201  switch (_frequency)
202  {
203  case Frequency1MHz:
204  default:
205  frequency = SPI_1_3125MHZ;
206  break;
207 
208  case Frequency2MHz:
209  frequency = SPI_2_625MHZ;
210  break;
211 
212  case Frequency4MHz:
213  frequency = SPI_5_25MHZ;
214  break;
215 
216  case Frequency8MHz:
217  frequency = SPI_10_5MHZ;
218  break;
219 
220  case Frequency16MHz:
221  frequency = SPI_21_0MHZ;
222  break;
223 
224  }
225  SPI.begin(frequency, bitOrder, dataMode);
226 #elif (RH_PLATFORM == RH_PLATFORM_RASPI) // Raspberry PI
227  uint8_t dataMode;
228  if (_dataMode == DataMode0)
229  dataMode = BCM2835_SPI_MODE0;
230  else if (_dataMode == DataMode1)
231  dataMode = BCM2835_SPI_MODE1;
232  else if (_dataMode == DataMode2)
233  dataMode = BCM2835_SPI_MODE2;
234  else if (_dataMode == DataMode3)
235  dataMode = BCM2835_SPI_MODE3;
236 
237  uint8_t bitOrder;
238  if (_bitOrder == BitOrderLSBFirst)
239  bitOrder = BCM2835_SPI_BIT_ORDER_LSBFIRST;
240  else
241  bitOrder = BCM2835_SPI_BIT_ORDER_MSBFIRST;
242 
243  uint32_t divider;
244  switch (_frequency)
245  {
246  case Frequency1MHz:
247  default:
248  divider = BCM2835_SPI_CLOCK_DIVIDER_256;
249  break;
250  case Frequency2MHz:
251  divider = BCM2835_SPI_CLOCK_DIVIDER_128;
252  break;
253  case Frequency4MHz:
254  divider = BCM2835_SPI_CLOCK_DIVIDER_64;
255  break;
256  case Frequency8MHz:
257  divider = BCM2835_SPI_CLOCK_DIVIDER_32;
258  break;
259  case Frequency16MHz:
260  divider = BCM2835_SPI_CLOCK_DIVIDER_16;
261  break;
262  }
263  SPI.begin(divider, bitOrder, dataMode);
264 #else
265  #warning RHHardwareSPI does not support this platform yet. Consider adding it and contributing a patch.
266 #endif
267 }
