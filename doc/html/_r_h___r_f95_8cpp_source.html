39 bool RH_RF95::init()
40 {
41 #if (RH_PLATFORM == RH_PLATFORM_RPI)
42  if(!RHLinuxSPI::init("/dev/spidev0.0"))
43 #else
44  if (!RHSPIDriver::init())
45 #endif
46  return false;
47 
48  // Determine the interrupt number that corresponds to the interruptPin
49 #if (RH_PLATFORM != RH_PLATFORM_RPI)
50  int interruptNumber = digitalPinToInterrupt(_interruptPin);
51  if (interruptNumber == NOT_AN_INTERRUPT)
52  return false;
53 #endif
54 
55  // No way to check the device type :-(
56 
57  // Set sleep mode, so we can also set LORA mode:
58  spiWrite(RH_RF95_REG_01_OP_MODE, RH_RF95_MODE_SLEEP | RH_RF95_LONG_RANGE_MODE);
59  delay(10); // Wait for sleep mode to take over from say, CAD
60  // Check we are in sleep mode, with LORA set
61  if (spiRead(RH_RF95_REG_01_OP_MODE) != (RH_RF95_MODE_SLEEP | RH_RF95_LONG_RANGE_MODE))
62  {
63 // Serial.println(spiRead(RH_RF95_REG_01_OP_MODE), HEX);
64  return false; // No device present?
65  }
66 
67  // Add by Adrien van den Bossche <vandenbo@univ-tlse2.fr> for Teensy
68  // ARM M4 requires the below. else pin interrupt doesn't work properly.
69  // On all other platforms, its innocuous, belt and braces
70 #if (RH_PLATFORM != RH_PLATFORM_RPI)
71  pinMode(_interruptPin, INPUT);
72 #endif
73 
74  // Set up interrupt handler
75  // Since there are a limited number of interrupt glue functions isr*() available,
76  // we can only support a limited number of devices simultaneously
77  // ON some devices, notably most Arduinos, the interrupt pin passed in is actuallt the
78  // interrupt number. You have to figure out the interruptnumber-to-interruptpin mapping
79  // yourself based on knwledge of what Arduino board you are running on.
80  if (_myInterruptIndex == 0xff)
81  {
82  // First run, no interrupt allocated yet
83  if (_interruptCount <= RH_RF95_NUM_INTERRUPTS)
84  _myInterruptIndex = _interruptCount++;
85  else
86  return false; // Too many devices, not enough interrupt vectors
87  }
88  _deviceForInterrupt[_myInterruptIndex] = this;
89 
90 #if (RH_PLATFORM == RH_PLATFORM_RPI)
91  switch( _myInterruptIndex ){
92  case 0:
93  attachInterrupt(_interruptPin, isr0, INT_EDGE_RISING);
94  break;
95  case 1:
96  attachInterrupt(_interruptPin, isr1, INT_EDGE_RISING);
97  break;
98  case 2:
99  attachInterrupt(_interruptPin, isr2, INT_EDGE_RISING);
100  break;
101  default:
102  return false;
103  }
104 #else
105  if (_myInterruptIndex == 0)
106  attachInterrupt(interruptNumber, isr0, RISING);
107  else if (_myInterruptIndex == 1)
108  attachInterrupt(interruptNumber, isr1, RISING);
109  else if (_myInterruptIndex == 2)
110  attachInterrupt(interruptNumber, isr2, RISING);
111  else
112  return false; // Too many devices, not enough interrupt vectors
113 #endif
114  // Set up FIFO
115  // We configure so that we can use the entire 256 byte FIFO for either receive
116  // or transmit, but not both at the same time
117  spiWrite(RH_RF95_REG_0E_FIFO_TX_BASE_ADDR, 0);
118  spiWrite(RH_RF95_REG_0F_FIFO_RX_BASE_ADDR, 0);
119 
120  // Packet format is preamble + explicit-header + payload + crc
121  // Explicit Header Mode
122  // payload is TO + FROM + ID + FLAGS + message data
123  // RX mode is implmented with RXCONTINUOUS
124  // max message data length is 255 - 4 = 251 octets
125 
126  setModeIdle();
127 
128  // Set up default configuration
129  // No Sync Words in LORA mode.
130  setModemConfig(Bw125Cr45Sf128); // Radio default
131 // setModemConfig(Bw125Cr48Sf4096); // slow and reliable?
132  setPreambleLength(8); // Default is 8
133  // An innocuous ISM frequency, same as RF22's
134  setFrequency(434.0);
135  // Lowish power
136  setTxPower(13);
137 
138  return true;
139 }

141 // C++ level interrupt handler for this instance
142 // LORA is unusual in that it has several interrupt lines, and not a single, combined one.
143 // On MiniWirelessLoRa, only one of the several interrupt lines (DI0) from the RFM95 is usefuly
144 // connnected to the processor.
145 // We use this to get RxDone and TxDone interrupts
146 void RH_RF95::handleInterrupt()
147 {
148 #if (RH_PLATFORM == RH_PLATFORM_RPI)
149 piLock(0);
150 #endif
151  // Read the interrupt register
152  uint8_t irq_flags = spiRead(RH_RF95_REG_12_IRQ_FLAGS);
153  if (_mode == RHModeRx && irq_flags & (RH_RF95_RX_TIMEOUT | RH_RF95_PAYLOAD_CRC_ERROR))
154  {
155  _rxBad++;
156  }
157  else if (_mode == RHModeRx && irq_flags & RH_RF95_RX_DONE)
158  {
159  // Have received a packet
160  uint8_t len = spiRead(RH_RF95_REG_13_RX_NB_BYTES);
161 
162  // Reset the fifo read ptr to the beginning of the packet
163  spiWrite(RH_RF95_REG_0D_FIFO_ADDR_PTR, spiRead(RH_RF95_REG_10_FIFO_RX_CURRENT_ADDR));
164  spiBurstRead(RH_RF95_REG_00_FIFO, _buf, len);
165  _bufLen = len;
166  spiWrite(RH_RF95_REG_12_IRQ_FLAGS, 0xff); // Clear all IRQ flags
167 
168  // Remember the RSSI of this packet
169  // this is according to the doc, but is it really correct?
170  // weakest receiveable signals are reported RSSI at about -66
171  _lastRssi = spiRead(RH_RF95_REG_1A_PKT_RSSI_VALUE) - 137;
172 
173  // We have received a message.
174  validateRxBuf();
175  if (_rxBufValid)
176  setModeIdle(); // Got one
177  }
178  else if (_mode == RHModeTx && irq_flags & RH_RF95_TX_DONE)
179  {
180  _txGood++;
181  setModeIdle();
182  }
183 
184  spiWrite(RH_RF95_REG_12_IRQ_FLAGS, 0xff); // Clear all IRQ flags
185 #if (RH_PLATFORM == RH_PLATFORM_RPI)
186 piUnlock(0);
187 #endif
188 }

209 // Check whether the latest received message is complete and uncorrupted
210 void RH_RF95::validateRxBuf()
211 {
212  if (_bufLen < 4)
213  return; // Too short to be a real message
214  // Extract the 4 headers
215  _rxHeaderTo = _buf[0];
216  _rxHeaderFrom = _buf[1];
217  _rxHeaderId = _buf[2];
218  _rxHeaderFlags = _buf[3];
219  if (_promiscuous ||
220  _rxHeaderTo == _thisAddress ||
221  _rxHeaderTo == RH_BROADCAST_ADDRESS)
222  {
223  _rxGood++;
224  _rxBufValid = true;
225  }

253 bool RH_RF95::recv(uint8_t* buf, uint8_t* len)
254 {
255  if (!available())
256  return false;
257  if (buf && len)
258  {
259  ATOMIC_BLOCK_START;
260  // Skip the 4 headers that are at the beginning of the rxBuf
261  if (*len > _bufLen-RH_RF95_HEADER_LEN)
262  *len = _bufLen-RH_RF95_HEADER_LEN;
263  memcpy(buf, _buf+RH_RF95_HEADER_LEN, *len);
264  ATOMIC_BLOCK_END;
265  }
266  clearRxBuf(); // This message accepted and cleared
267  return true;
268 }

270 bool RH_RF95::send(const uint8_t* data, uint8_t len)
271 {
272  if (len > RH_RF95_MAX_MESSAGE_LEN)
273  return false;
274 
275  waitPacketSent(); // Make sure we dont interrupt an outgoing message
276  setModeIdle();
277 
278  // Position at the beginning of the FIFO
279  spiWrite(RH_RF95_REG_0D_FIFO_ADDR_PTR, 0);
280  // The headers
281  spiWrite(RH_RF95_REG_00_FIFO, _txHeaderTo);
282  spiWrite(RH_RF95_REG_00_FIFO, _txHeaderFrom);
283  spiWrite(RH_RF95_REG_00_FIFO, _txHeaderId);
284  spiWrite(RH_RF95_REG_00_FIFO, _txHeaderFlags);
285  // The message data
286  spiBurstWrite(RH_RF95_REG_00_FIFO, data, len);
287  spiWrite(RH_RF95_REG_22_PAYLOAD_LENGTH, len + RH_RF95_HEADER_LEN);
288 
289  setModeTx(); // Start the transmitter
290  // when Tx is done, interruptHandler will fire and radio mode will return to STANDBY
291  return true;
292 }

369 void RH_RF95::setTxPower(int8_t power)
370 {
371  if (power > 23)
372  power = 23;
373  if (power < 5)
374  power = 5;
375 
376  // For RH_RF95_PA_DAC_ENABLE, manual says '+20dBm on PA_BOOST when OutputPower=0xf'
377  // RH_RF95_PA_DAC_ENABLE actually adds about 3dBm to all power levels. We will us it
378  // for 21, 22 and 23dBm
379  if (power > 20)
380  {
381  spiWrite(RH_RF95_REG_4D_PA_DAC, RH_RF95_PA_DAC_ENABLE);
382  power -= 3;
383  }
384  else
385  {
386  spiWrite(RH_RF95_REG_4D_PA_DAC, RH_RF95_PA_DAC_DISABLE);
387  }
388 
389  // RFM95/96/97/98 does not have RFO pins connected to anything. Only PA_BOOST
390  // pin is connected, so must use PA_BOOST
391  // Pout = 2 + OutputPower.
392  // The documentation is pretty confusing on this topic: PaSelect says the max power is 20dBm,
393  // but OutputPower claims it would be 17dBm.
394  // My measurements show 20dBm is correct
395  spiWrite(RH_RF95_REG_09_PA_CONFIG, RH_RF95_PA_SELECT | (power-5));
396 }
