15 // Caution: on Arduino Uno and many other CPUs, digitalWrite is quite slow, taking about 4us
16 // digitalWrite is also slow, taking about 3.5us
17 // resulting in very slow SPI bus speeds using this technique, up to about 120us per octet of transfer
18 uint8_t RHSoftwareSPI::transfer(uint8_t data)
19 {
20  uint8_t readData;
21  uint8_t writeData;
22  uint8_t builtReturn;
23  uint8_t mask;
24 
25  if (_bitOrder == BitOrderMSBFirst)
26  {
27  mask = 0x80;
28  }
29  else
30  {
31  mask = 0x01;
32  }
33  builtReturn = 0;
34  readData = 0;
35 
36  for (uint8_t count=0; count<8; count++)
37  {
38  if (data & mask)
39  {
40  writeData = HIGH;
41  }
42  else
43  {
44  writeData = LOW;
45  }
46 
47  if (_clockPhase == 1)
48  {
49  // CPHA=1, miso/mosi changing state now
50  digitalWrite(_mosi, writeData);
51  digitalWrite(_sck, ~_clockPolarity);
52  delayPeriod();
53 
54  // CPHA=1, miso/mosi stable now
55  readData = digitalRead(_miso);
56  digitalWrite(_sck, _clockPolarity);
57  delayPeriod();
58  }
59  else
60  {
61  // CPHA=0, miso/mosi changing state now
62  digitalWrite(_mosi, writeData);
63  digitalWrite(_sck, _clockPolarity);
64  delayPeriod();
65 
66  // CPHA=0, miso/mosi stable now
67  readData = digitalRead(_miso);
68  digitalWrite(_sck, ~_clockPolarity);
69  delayPeriod();
70  }
71 
72  if (_bitOrder == BitOrderMSBFirst)
73  {
74  mask >>= 1;
75  builtReturn |= (readData << (7 - count));
76  }
77  else
78  {
79  mask <<= 1;
80  builtReturn |= (readData << count);
81  }
82  }
83 
84  digitalWrite(_sck, _clockPolarity);
85 
86  return builtReturn;
87 }
