44 void RHRouter::addRouteTo(uint8_t dest, uint8_t next_hop, uint8_t state)
45 {
46  uint8_t i;
47 
48  // First look for an existing entry we can update
49  for (i = 0; i < RH_ROUTING_TABLE_SIZE; i++)
50  {
51  if (_routes[i].dest == dest)
52  {
53  _routes[i].dest = dest;
54  _routes[i].next_hop = next_hop;
55  _routes[i].state = state;
56  return;
57  }
58  }
59 
60  // Look for an invalid entry we can use
61  for (i = 0; i < RH_ROUTING_TABLE_SIZE; i++)
62  {
63  if (_routes[i].state == Invalid)
64  {
65  _routes[i].dest = dest;
66  _routes[i].next_hop = next_hop;
67  _routes[i].state = state;
68  return;
69  }
70  }
71 
72  // Need to make room for a new one
73  retireOldestRoute();
74  // Should be an invalid slot now
75  for (i = 0; i < RH_ROUTING_TABLE_SIZE; i++)
76  {
77  if (_routes[i].state == Invalid)
78  {
79  _routes[i].dest = dest;
80  _routes[i].next_hop = next_hop;
81  _routes[i].state = state;
82  }
83  }
84 }

160 // Waits for delivery to the next hop (but not for delivery to the final destination)
161 uint8_t RHRouter::sendtoFromSourceWait(uint8_t* buf, uint8_t len, uint8_t dest, uint8_t source, uint8_t flags)
162 {
163  if (((uint16_t)len + sizeof(RoutedMessageHeader)) > _driver.maxMessageLength())
164  return RH_ROUTER_ERROR_INVALID_LENGTH;
165 
166  // Construct a RH RouterMessage message
167  _tmpMessage.header.source = source;
168  _tmpMessage.header.dest = dest;
169  _tmpMessage.header.hops = 0;
170  _tmpMessage.header.id = _lastE2ESequenceNumber++;
171  _tmpMessage.header.flags = flags;
172  memcpy(_tmpMessage.data, buf, len);
173 
174  return route(&_tmpMessage, sizeof(RoutedMessageHeader)+len);
175 }

178 uint8_t RHRouter::route(RoutedMessage* message, uint8_t messageLen)
179 {
180  // Reliably deliver it if possible. See if we have a route:
181  uint8_t next_hop = RH_BROADCAST_ADDRESS;
182  if (message->header.dest != RH_BROADCAST_ADDRESS)
183  {
184  RoutingTableEntry* route = getRouteTo(message->header.dest);
185  if (!route)
186  return RH_ROUTER_ERROR_NO_ROUTE;
187  next_hop = route->next_hop;
188  }
189 
190  if (!RHReliableDatagram::sendtoWait((uint8_t*)message, messageLen, next_hop))
191  return RH_ROUTER_ERROR_UNABLE_TO_DELIVER;
192 
193  return RH_ROUTER_ERROR_NONE;
194 }

204 bool RHRouter::recvfromAck(uint8_t* buf, uint8_t* len, uint8_t* source, uint8_t* dest, uint8_t* id, uint8_t* flags)
205 {
206  uint8_t tmpMessageLen = sizeof(_tmpMessage);
207  uint8_t _from;
208  uint8_t _to;
209  uint8_t _id;
210  uint8_t _flags;
211  if (RHReliableDatagram::recvfromAck((uint8_t*)&_tmpMessage, &tmpMessageLen, &_from, &_to, &_id, &_flags))
212  {
213  // Here we simulate networks with limited visibility between nodes
214  // so we can test routing
215 #ifdef RH_TEST_NETWORK
216  if (
217 #if RH_TEST_NETWORK==1
218  // This network looks like 1-2-3-4
219  (_thisAddress == 1 && _from == 2)
220  || (_thisAddress == 2 && (_from == 1 || _from == 3))
221  || (_thisAddress == 3 && (_from == 2 || _from == 4))
222  || (_thisAddress == 4 && _from == 3)
223 
224 #elif RH_TEST_NETWORK==2
225  // This network looks like 1-2-4
226  // | | |
227  // --3--
228  (_thisAddress == 1 && (_from == 2 || _from == 3))
229  || _thisAddress == 2
230  || _thisAddress == 3
231  || (_thisAddress == 4 && (_from == 2 || _from == 3))
232 
233 #elif RH_TEST_NETWORK==3
234  // This network looks like 1-2-4
235  // | |
236  // --3--
237  (_thisAddress == 1 && (_from == 2 || _from == 3))
238  || (_thisAddress == 2 && (_from == 1 || _from == 4))
239  || (_thisAddress == 3 && (_from == 1 || _from == 4))
240  || (_thisAddress == 4 && (_from == 2 || _from == 3))
241 
242 #elif RH_TEST_NETWORK==4
243  // This network looks like 1-2-3
244  // |
245  // 4
246  (_thisAddress == 1 && _from == 2)
247  || _thisAddress == 2
248  || (_thisAddress == 3 && _from == 2)
249  || (_thisAddress == 4 && _from == 2)
250 
251 #endif
252 )
253  {
254  // OK
255  }
256  else
257  {
258  return false; // Pretend we got nothing
259  }
260 #endif
261 
262  peekAtMessage(&_tmpMessage, tmpMessageLen);
263  // See if its for us or has to be routed
264  if (_tmpMessage.header.dest == _thisAddress || _tmpMessage.header.dest == RH_BROADCAST_ADDRESS)
265  {
266  // Deliver it here
267  if (source) *source = _tmpMessage.header.source;
268  if (dest) *dest = _tmpMessage.header.dest;
269  if (id) *id = _tmpMessage.header.id;
270  if (flags) *flags = _tmpMessage.header.flags;
271  uint8_t msgLen = tmpMessageLen - sizeof(RoutedMessageHeader);
272  if (*len > msgLen)
273  *len = msgLen;
274  memcpy(buf, _tmpMessage.data, *len);
275  return true; // Its for you!
276  }
277  else if ( _tmpMessage.header.dest != RH_BROADCAST_ADDRESS
278  && _tmpMessage.header.hops++ < _max_hops)
279  {
280  // Maybe it has to be routed to the next hop
281  // REVISIT: if it fails due to no route or unable to deliver to the next hop,
282  // tell the originator. BUT HOW?
283  route(&_tmpMessage, tmpMessageLen);
284  }
285  // Discard it and maybe wait for another
286  }
287  return false;
288 }
