17 bool RH_NRF24::init()
18 {
19  // Teensy with nRF24 is unreliable at 8MHz:
20  // so is Arduino with RF73
21  _spi.setFrequency(RHGenericSPI::Frequency1MHz);
22  if (!RHNRFSPIDriver::init())
23  return false;
24 
25  // Initialise the slave select pin
26  pinMode(_chipEnablePin, OUTPUT);
27  digitalWrite(_chipEnablePin, LOW);
28 
29  // Clear interrupts
30  spiWriteRegister(RH_NRF24_REG_07_STATUS, RH_NRF24_RX_DR | RH_NRF24_TX_DS | RH_NRF24_MAX_RT);
31  // Enable dynamic payload length on all pipes
32  spiWriteRegister(RH_NRF24_REG_1C_DYNPD, RH_NRF24_DPL_ALL);
33  // Enable dynamic payload length, disable payload-with-ack, enable noack
34  spiWriteRegister(RH_NRF24_REG_1D_FEATURE, RH_NRF24_EN_DPL | RH_NRF24_EN_DYN_ACK);
35  // Test if there is actually a device connected and responding
36  // CAUTION: RFM73 and version 2.0 silicon may require ACTIVATE
37  if (spiReadRegister(RH_NRF24_REG_1D_FEATURE) != (RH_NRF24_EN_DPL | RH_NRF24_EN_DYN_ACK))
38  {
39  spiWrite(RH_NRF24_COMMAND_ACTIVATE, 0x73);
40  // Enable dynamic payload length, disable payload-with-ack, enable noack
41  spiWriteRegister(RH_NRF24_REG_1D_FEATURE, RH_NRF24_EN_DPL | RH_NRF24_EN_DYN_ACK);
42  if (spiReadRegister(RH_NRF24_REG_1D_FEATURE) != (RH_NRF24_EN_DPL | RH_NRF24_EN_DYN_ACK))
43  return false;
44  }
45 
46  // Make sure we are powered down
47  setModeIdle();
48 
49  // Flush FIFOs
50  flushTx();
51  flushRx();
52 
53  setChannel(2); // The default, in case it was set by another app without powering down
54  setRF(RH_NRF24::DataRate2Mbps, RH_NRF24::TransmitPower0dBm);
55 
56  return true;
57 }

108 bool RH_NRF24::setNetworkAddress(uint8_t* address, uint8_t len)
109 {
110  if (len < 3 || len > 5)
111  return false;
112 
113  // Set both TX_ADDR and RX_ADDR_P0 for auto-ack with Enhanced shockwave
114  spiWriteRegister(RH_NRF24_REG_03_SETUP_AW, len-2); // Mapping [3..5] = [1..3]
115  spiBurstWriteRegister(RH_NRF24_REG_0A_RX_ADDR_P0, address, len);
116  spiBurstWriteRegister(RH_NRF24_REG_10_TX_ADDR, address, len);
117  return true;

120 bool RH_NRF24::setRF(DataRate data_rate, TransmitPower power)
121 {
122  uint8_t value = (power << 1) & RH_NRF24_PWR;
123  // Ugly mapping of data rates to noncontiguous 2 bits:
124  if (data_rate == DataRate250kbps)
125  value |= RH_NRF24_RF_DR_LOW;
126  else if (data_rate == DataRate2Mbps)
127  value |= RH_NRF24_RF_DR_HIGH;
128  // else DataRate1Mbps, 00
129 
130  // RFM73 needs this:
131  value |= RH_NRF24_LNA_HCURR;
132 
133  spiWriteRegister(RH_NRF24_REG_06_RF_SETUP, value);
134  // If we were using auto-ack, we would have to set the appropriate timeout in reg 4 here
135  // see NRF24::setRF()
136  return true;
137 }

169 void RH_NRF24::setModeTx()
170 {
171  if (_mode != RHModeTx)
172  {
173  // Its the CE rising edge that puts us into TX mode
174  // CE staying high makes us go to standby-II when the packet is sent
175  digitalWrite(_chipEnablePin, LOW);

184 bool RH_NRF24::send(const uint8_t* data, uint8_t len)
185 {
186  if (len > RH_NRF24_MAX_MESSAGE_LEN)
187  return false;
188  // Set up the headers
189  _buf[0] = _txHeaderTo;
190  _buf[1] = _txHeaderFrom;
191  _buf[2] = _txHeaderId;
192  _buf[3] = _txHeaderFlags;
193  memcpy(_buf+RH_NRF24_HEADER_LEN, data, len);
194  spiBurstWrite(RH_NRF24_COMMAND_W_TX_PAYLOAD_NOACK, _buf, len + RH_NRF24_HEADER_LEN);
195  setModeTx();
196  // Radio will return to Standby II mode after transmission is complete
197  _txGood++;
198  return true;
199 }

201 bool RH_NRF24::waitPacketSent()
202 {
203  // If we are not currently in transmit mode, there is no packet to wait for
204  if (_mode != RHModeTx)
205  return false;
206 
207  // Wait for either the Data Sent or Max ReTries flag, signalling the
208  // end of transmission
209  // We dont actually use auto-ack, so prob dont expect to see RH_NRF24_MAX_RT
210  uint8_t status;
211  while (!((status = statusRead()) & (RH_NRF24_TX_DS | RH_NRF24_MAX_RT)))
212  YIELD;
213 
214  // Must clear RH_NRF24_MAX_RT if it is set, else no further comm
215  if (status & RH_NRF24_MAX_RT)
216  flushTx();
217  setModeIdle();
218  spiWriteRegister(RH_NRF24_REG_07_STATUS, RH_NRF24_TX_DS | RH_NRF24_MAX_RT);
219  // Return true if data sent, false if MAX_RT
220  return status & RH_NRF24_TX_DS;
221 }

229 bool RH_NRF24::printRegisters()
230 {
231 #ifdef RH_HAVE_SERIAL
232  // Iterate over register range, but don't process registers not in use.
233  for (uint8_t r = RH_NRF24_REG_00_CONFIG; r <= RH_NRF24_REG_1D_FEATURE; r++)
234  {
235  if ((r <= RH_NRF24_REG_17_FIFO_STATUS) || (r >= RH_NRF24_REG_1C_DYNPD))
236  {
237  Serial.print(r, HEX);
238  Serial.print(": ");
239  uint8_t len = 1;
240  // Address registers are 5 bytes in size
241  if ( (RH_NRF24_REG_0A_RX_ADDR_P0 == r)
242  || (RH_NRF24_REG_0B_RX_ADDR_P1 == r)
243  || (RH_NRF24_REG_10_TX_ADDR == r) )
244  {
245  len = 5;
246  }
247  uint8_t buf[5];
248  spiBurstReadRegister(r, buf, len);
249  for (uint8_t j = 0; j < len; ++j)
250  {
251  Serial.print(buf[j], HEX);
252  Serial.print(" ");
253  }
254  Serial.println("");
255  }
256  }
257 #endif
258 
259  return true;
260 }

264 {
265  if (_bufLen < 4)

281 bool RH_NRF24::available()
282 {
283  if (!_rxBufValid)
284  {
285  if (_mode == RHModeTx)
286  return false;
287  setModeRx();
288  if (spiReadRegister(RH_NRF24_REG_17_FIFO_STATUS) & RH_NRF24_RX_EMPTY)
289  return false;
290  // Manual says that messages > 32 octets should be discarded
291  uint8_t len = spiRead(RH_NRF24_COMMAND_R_RX_PL_WID);
292  if (len > 32)
293  {
294  flushRx();
295  clearRxBuf();
296  setModeIdle();
297  return false;
298  }
299  // Clear read interrupt
300  spiWriteRegister(RH_NRF24_REG_07_STATUS, RH_NRF24_RX_DR);
301  // Get the message into the RX buffer, so we can inspect the headers
302  spiBurstRead(RH_NRF24_COMMAND_R_RX_PAYLOAD, _buf, len);
303  _bufLen = len;
304  // 140 microsecs (32 octet payload)
305  validateRxBuf();
306  if (_rxBufValid)
307  setModeIdle(); // Got one
308  }
309  return _rxBufValid;
310 }

318 bool RH_NRF24::recv(uint8_t* buf, uint8_t* len)
319 {
320  if (!available())
321  return false;
322  if (buf && len)
323  {
324  // Skip the 4 headers that are at the beginning of the rxBuf
325  if (*len > _bufLen-RH_NRF24_HEADER_LEN)
326  *len = _bufLen-RH_NRF24_HEADER_LEN;
327  memcpy(buf, _buf+RH_NRF24_HEADER_LEN, *len);
328  }
329  clearRxBuf(); // This message accepted and cleared
330  return true;
331 }
