37 bool RH_TCP::connectToServer()
38 {
39  struct addrinfo hints;
40  struct addrinfo *result, *rp;
41  int sfd, s;
42  struct sockaddr_storage peer_addr;
43  socklen_t peer_addr_len;
44 
45  memset(&hints, 0, sizeof(struct addrinfo));
46  hints.ai_family = AF_UNSPEC; // Allow IPv4 or IPv6
47  hints.ai_socktype = SOCK_STREAM; // Stream socket
48  hints.ai_flags = AI_PASSIVE; // For wildcard IP address
49  hints.ai_protocol = 0; // Any protocol
50  hints.ai_canonname = NULL;
51  hints.ai_addr = NULL;
52  hints.ai_next = NULL;
53 
54  std::string server(_server);
55  std::string port("4000");
56  size_t indexOfSeparator = server.find_first_of(':');
57  if (indexOfSeparator != std::string::npos)
58  {
59  port = server.substr(indexOfSeparator+1);
60  server.erase(indexOfSeparator);
61  }
62 
63  s = getaddrinfo(server.c_str(), port.c_str(), &hints, &result);
64  if (s != 0)
65  {
66  fprintf(stderr, "RH_TCP::connect getaddrinfo failed: %s\n", gai_strerror(s));
67  return false;
68  }
69 
70  // getaddrinfo() returns a list of address structures.
71  // Try each address until we successfully connect(2).
72  // If socket(2) (or connect(2)) fails, we (close the socket
73  // and) try the next address. */
74 
75  for (rp = result; rp != NULL; rp = rp->ai_next)
76  {
77  _socket = socket(rp->ai_family, rp->ai_socktype, rp->ai_protocol);
78  if (_socket == -1)
79  continue;
80 
81  if (connect(_socket, rp->ai_addr, rp->ai_addrlen) == 0)
82  break; /* Success */
83 
84  close(_socket);
85  }
86 
87  if (rp == NULL)
88  { /* No address succeeded */
89  fprintf(stderr, "RH_TCP::connect could not connect to %s\n", _server);
90  return false;
91  }
92 
93  freeaddrinfo(result); /* No longer needed */
94 
95  // Now make the socket non-blocking
96  int on = 1;
97  int rc = ioctl(_socket, FIONBIO, (char *)&on);
98  if (rc < 0)
99  {
100  fprintf(stderr,"RH_TCP::init failed to set socket non-blocking: %s\n", strerror(errno));
101  close(_socket);
102  _socket = -1;
103  return false;
104  }
105  return true;
106 }

114 void RH_TCP::checkForEvents()
115 {
116  #define RH_TCP_SOCKETBUF_LEN 500
117  static uint8_t socketBuf[RH_TCP_SOCKETBUF_LEN]; // Room for several messages
118  static uint16_t socketBufLen = 0;
119 
120  // Read at most the amount of space we have left in the buffer
121  ssize_t count = read(_socket, socketBuf + socketBufLen, sizeof(socketBuf) - socketBufLen);
122  if (count < 0)
123  {
124  if (errno != EAGAIN)
125  {
126  fprintf(stderr,"RH_TCP::checkForEvents read error: %s\n", strerror(errno));
127  exit(1);
128  }
129  }
130  else if (count == 0)
131  {
132  // End of file
133  fprintf(stderr,"RH_TCP::checkForEvents unexpected end of file on read\n");
134  exit(1);
135  }
136  else
137  {
138  socketBufLen += count;
139  while (socketBufLen >= 5)
140  {
141  RHTcpTypeMessage* message = ((RHTcpTypeMessage*)socketBuf);
142  uint32_t len = ntohl(message->length);
143  uint32_t messageLen = len + sizeof(message->length);
144  if (len > sizeof(socketBuf) - sizeof(message->length))
145  {
146  // Bogus length
147  fprintf(stderr, "RH_TCP::checkForEvents read ridiculous length: %d. Corrupt message stream? Aborting\n", len);
148  exit(1);
149  }
150  if (socketBufLen >= len + sizeof(message->length))
151  {
152  // Got at least all of this message
153  if (message->type == RH_TCP_MESSAGE_TYPE_PACKET && len >= 5)
154  {
155  // REVISIT: need to check if we are actually receiving?
156  // Its a new packet, extract the headers and payload
157  RHTcpPacket* packet = ((RHTcpPacket*)socketBuf);
158  _rxHeaderTo = packet->to;
159  _rxHeaderFrom = packet->from;
160  _rxHeaderId = packet->id;
161  _rxHeaderFlags = packet->flags;
162  uint32_t payloadLen = len - 5;
163  if (payloadLen <= sizeof(_rxBuf))
164  {
165  // Enough room in our receiver buffer
166  memcpy(_rxBuf, packet->payload, payloadLen);
167  _rxBufLen = payloadLen;
168  _rxBufFull = true;
169  }
170  }
171  // check for other message types here
172  // Now remove the used message by copying the trailing bytes (maybe start of a new message?)
173  // to the top of the buffer
174  memcpy(socketBuf, socketBuf + messageLen, sizeof(socketBuf) - messageLen);
175  socketBufLen -= messageLen;
176  }
177  }
178  }
179 }

214 {
215  int max_fd;
216  fd_set input;
217  int result;
218 
219  FD_ZERO(&input);
220  FD_SET(_socket, &input);
221  max_fd = _socket + 1;
222 
223  if (timeout)
224  {
225  struct timeval timer;
226  // Timeout is in milliseconds
227  timer.tv_sec = timeout / 1000;
228  timer.tv_usec = (timeout % 1000) * 1000;
229  result = select(max_fd, &input, NULL, NULL, &timer);
230  }
231  else
232  {
233  result = select(max_fd, &input, NULL, NULL, NULL);
234  }
235  if (result < 0)
236  fprintf(stderr, "RH_TCP::waitAvailableTimeout: select failed %s\n", strerror(errno));
237  return result > 0;
